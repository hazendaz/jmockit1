7   /**
8    * Utility class for creation of {@link Vet} data in the test database, to be used in integration tests.
9    */
10  public final class VetData extends TestDatabase {
11      @NonNull
12      public Vet create(@NonNull String fullName, @NonNull String... specialtyNames) {
13          String[] names = fullName.split(" ");
14  
15          Vet vet = new Vet();
16          vet.setFirstName(names[0]);
17          vet.setLastName(names[names.length - 1]);
18  
19          for (String specialtyName : specialtyNames) {
20              Specialty specialty = new Specialty();
21              specialty.setName(specialtyName);
22  
23              vet.getSpecialties().add(specialty);
24          }
25  
26          db.save(vet);
27          return vet;
28      }
29  }
