6   package petclinic.visits;
7   
8   import jakarta.persistence.Entity;
9   import jakarta.persistence.JoinColumn;
10  import jakarta.persistence.ManyToOne;
11  import jakarta.persistence.Temporal;
12  import jakarta.persistence.TemporalType;
13  import jakarta.validation.constraints.NotNull;
14  
15  import java.util.Date;
16  
17  import petclinic.pets.Pet;
18  import petclinic.util.BaseEntity;
19  
20  /**
21   * A visit from a pet and its owner to the clinic.
22   */
23  @Entity
24  public class Visit extends BaseEntity {
25      private static final long serialVersionUID = 1L;
26  
27      @NotNull
28      @Temporal(TemporalType.TIMESTAMP)
29      private Date date;
30  
31      @NotNull
32      private String description;
33  
34      @NotNull
35      @ManyToOne
36      @JoinColumn(name = "petId")
37      private Pet pet;
38  
39      /**
40       * Creates a new instance of Visit for the current date.
41       */
42      public Visit() {
43          date = new Date();
44      }
45  
46      public Date getDate() {
47          return date;
48      }
49  
50      public void setDate(Date date) {
51          this.date = date;
52      }
53  
54      public String getDescription() {
55          return description;
56      }
57  
58      public void setDescription(String description) {
59          this.description = description;
60      }
61  
62      public Pet getPet() {
63          return pet;
64      }
65  
66      public void setPet(Pet pet) {
67          this.pet = pet;
68      }
69  }
