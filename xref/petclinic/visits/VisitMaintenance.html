3   import edu.umd.cs.findbugs.annotations.NonNull;
4   import edu.umd.cs.findbugs.annotations.Nullable;

6   import java.util.List;

8   import javax.inject.Inject;
9   import javax.transaction.Transactional;
10  
11  import petclinic.pets.Pet;
12  import petclinic.util.Database;
13  
14  /**
15   * A domain service class for {@link Visit}-related business operations.
16   */
17  @Transactional
18  public class VisitMaintenance {
19      @Inject
20      private Database db;
21  
22      public void create(@NonNull Pet visitedPet, @NonNull Visit visitData) {
23          visitData.setPet(visitedPet);
24          visitedPet.addVisit(visitData);
25          db.save(visitData);
26      }
27  
28      @Nullable
29      public Visit findById(int visitId) {
30          return db.findById(Visit.class, visitId);
31      }
32  
33      @NonNull
34      public List<Visit> findByPetId(int petId) {
35          return db.find("select v from Visit v where v.pet.id = ?1", petId);
36      }
37  }
