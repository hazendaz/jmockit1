6   package petclinic.vets;
7   
8   import edu.umd.cs.findbugs.annotations.Nullable;
9   
10  import jakarta.faces.view.ViewScoped;
11  import jakarta.inject.Inject;
12  import jakarta.inject.Named;
13  import jakarta.transaction.Transactional;
14  
15  import java.util.List;
16  
17  /**
18   * An application service class that handles {@link Vet}-related operations from the vet screen.
19   */
20  @Named
21  @Transactional
22  @ViewScoped
23  public class VetScreen {
24      @Inject
25      private VetMaintenance vetMaintenance;
26      @Nullable
27      private List<Vet> vets;
28  
29      @Nullable
30      public List<Vet> getVets() {
31          return vets;
32      }
33  
34      public void showVetList() {
35          vets = vetMaintenance.findAll();
36      }
37  }
