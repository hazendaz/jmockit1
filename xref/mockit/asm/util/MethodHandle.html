1   package mockit.asm.util;

3   import javax.annotation.Nonnegative;
4   import javax.annotation.Nonnull;
5   
6   /**
7    * A reference to a method.
8    */
9   public final class MethodHandle {
10      public interface Tag {
11          int TAG_GETFIELD = 1;
12          int TAG_GETSTATIC = 2;
13          int TAG_PUTFIELD = 3;
14          int TAG_PUTSTATIC = 4;
15          int TAG_INVOKEVIRTUAL = 5;
16          int TAG_INVOKESTATIC = 6;
17          int TAG_INVOKESPECIAL = 7;
18          int TAG_NEWINVOKESPECIAL = 8;
19          int TAG_INVOKEINTERFACE = 9;
20      }
21  
22      /**
23       * The kind of method designated by this handle. Should be one of the {@link Tag Tag} constants.
24       */
25      @Nonnegative
26      public final int tag;
27  
28      /**
29       * The internal name of the class that owns the method designated by this handle.
30       */
31      @Nonnull
32      public final String owner;
33  
34      /**
35       * The name of the method designated by this handle.
36       */
37      @Nonnull
38      public final String name;
39  
40      /**
41       * The descriptor of the method designated by this handle.
42       */
43      @Nonnull
44      public final String desc;
45  
46      /**
47       * Initializes a new method handle.
48       *
49       * @param tag
50       *            the kind of method designated by this handle. Must be one of the {@link Tag} constants.
51       * @param owner
52       *            the internal name of the class that owns the method designated by this handle.
53       * @param name
54       *            the name of the method designated by this handle.
55       * @param desc
56       *            the descriptor of the method designated by this handle.
57       */
58      public MethodHandle(@Nonnegative int tag, @Nonnull String owner, @Nonnull String name, @Nonnull String desc) {
59          this.tag = tag;
60          this.owner = owner;
61          this.name = name;
62          this.desc = desc;
63      }
64  
65      @Override
66      public boolean equals(Object obj) {
67          if (obj == this) {
68              return true;
69          }
70  
71          if (!(obj instanceof MethodHandle)) {
72              return false;
73          }
74  
75          MethodHandle h = (MethodHandle) obj;
76          return tag == h.tag && owner.equals(h.owner) && name.equals(h.name) && desc.equals(h.desc);
77      }
78  
79      @Override
80      public int hashCode() {
81          return tag + owner.hashCode() * name.hashCode() * desc.hashCode();
82      }
83  }
