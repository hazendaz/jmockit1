5   package mockit.coverage.reporting;
6   
7   import edu.umd.cs.findbugs.annotations.NonNull;
8   
9   import java.io.File;
10  import java.io.IOException;
11  import java.io.PrintWriter;
12  import java.nio.charset.StandardCharsets;
13  import java.util.regex.Pattern;
14  
15  public final class OutputFile extends PrintWriter {
16      private static final Pattern PATH_SEPARATOR = Pattern.compile("/");
17  
18      @NonNull
19      private final String relPathToOutDir;
20      private final boolean sourceFile;
21  
22      public OutputFile(@NonNull File file) throws IOException {
23          super(file, StandardCharsets.UTF_8);
24          relPathToOutDir = "";
25          sourceFile = false;
26      }
27  
28      public OutputFile(@NonNull String outputDir, @NonNull String sourceFilePath) throws IOException {
29          super(getOutputFileCreatingDirIfNeeded(outputDir, sourceFilePath));
30          relPathToOutDir = getRelativeSubPathToOutputDir(sourceFilePath);
31          sourceFile = true;
32      }
33  
34      @NonNull
35      private static File getOutputFileCreatingDirIfNeeded(@NonNull String outputDir, @NonNull String sourceFilePath) {
36          File outputFile = getOutputFile(outputDir, sourceFilePath);
37          File parentDir = outputFile.getParentFile();
38  
39          if (!parentDir.exists()) {
40              boolean outputDirCreated = parentDir.mkdirs();
41              assert outputDirCreated : "Failed to create output dir: " + outputDir;
42          }
43  
44          return outputFile;
45      }
46  
47      @NonNull
48      static File getOutputFile(@NonNull String outputDir, @NonNull String sourceFilePath) {
49          int p = sourceFilePath.lastIndexOf('.');
50          String outputFileName = sourceFilePath.substring(0, p) + ".html";
51          return new File(outputDir, outputFileName);
52      }
53  
54      @NonNull
55      private static String getRelativeSubPathToOutputDir(@NonNull String filePath) {
56          StringBuilder cssRelPath = new StringBuilder();
57          int n = PATH_SEPARATOR.split(filePath).length;
58  
59          for (int i = 1; i < n; i++) {
60              cssRelPath.append("../");
61          }
62  
63          return cssRelPath.toString();
64      }
65  
66      public void writeCommonHeader(@NonNull String pageTitle) {
67          println("<!DOCTYPE html>");
68          println("<html>");
69          println("<head>");
70          println("  <title>" + pageTitle + "</title>");
71          println("  <meta charset='UTF-8'>");
72          println("  <link rel='stylesheet' type='text/css' href='" + relPathToOutDir + (sourceFile ? "source" : "index")
73                  + ".css'>");
74          println("  <link rel='shortcut icon' type='image/png' href='" + relPathToOutDir + "logo.png'>");
75          println("  <script src='" + relPathToOutDir + "coverage.js'></script>");
76          println("  <base target='_blank'>");
77  
78          if (sourceFile) {
79              println("  <script src='" + relPathToOutDir + "prettify.js'></script>");
80          }
81  
82          println("</head>");
83          println(sourceFile ? "<body onload='prettyPrint()'>" : "<body>");
84      }
85  
86      public void writeCommonFooter() {
87          println("</body>");
88          println("</html>");
89      }
90  }
