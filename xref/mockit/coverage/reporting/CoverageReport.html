5   package mockit.coverage.reporting;
6   
7   import edu.umd.cs.findbugs.annotations.NonNull;
8   import edu.umd.cs.findbugs.annotations.Nullable;
9   
10  import java.io.File;
11  import java.io.IOException;
12  import java.util.ArrayList;
13  import java.util.Collection;
14  import java.util.HashMap;
15  import java.util.List;
16  import java.util.Map;
17  import java.util.Map.Entry;
18  import java.util.Set;
19  
20  import mockit.coverage.Configuration;
21  import mockit.coverage.data.CoverageData;
22  import mockit.coverage.data.FileCoverageData;
23  import mockit.coverage.reporting.packages.IndexPage;
24  import mockit.coverage.reporting.sourceFiles.FileCoverageReport;
25  import mockit.coverage.reporting.sourceFiles.InputFile;
26  
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  public final class CoverageReport {
31  
32      /** The logger. */
33      private static final Logger logger = LoggerFactory.getLogger(CoverageReport.class);
34  
35      @NonNull
36      private final String outputDir;
37      private boolean outputDirCreated;
38      @Nullable
39      private final List<File> sourceDirs;
40      @NonNull
41      private final Map<String, FileCoverageData> fileToFileData;
42      @NonNull
43      private final Map<String, List<String>> packageToFiles;
44      private final boolean withCallPoints;
45      @Nullable
46      private final Collection<String> sourceFilesNotFound;
47  
48      public CoverageReport(@NonNull String outputDir, boolean outputDirCreated, @Nullable String[] srcDirs,
49              @NonNull CoverageData coverageData, boolean withCallPoints) {
50          this.outputDir = Configuration.getOrChooseOutputDirectory(outputDir, "coverage-report");
51          this.outputDirCreated = outputDirCreated;
52          sourceDirs = srcDirs == null ? null : new SourceFiles().buildListOfSourceDirectories(srcDirs);
53          fileToFileData = coverageData.getFileToFileData();
54          packageToFiles = new HashMap<>();
55          this.withCallPoints = withCallPoints;
56          sourceFilesNotFound = srcDirs == null ? null : new ArrayList<>();
57      }
58  
59      public void generate() throws IOException {
60          createReportOutputDirIfNotExists();
61  
62          File outputFile = createOutputFileForIndexPage();
63  
64          if (outputFile == null) {
65              return;
66          }
67  
68          boolean withSourceFilePages = sourceDirs != null;
69  
70          if (withSourceFilePages && sourceDirs.size() > 1) {
71              logger.info("JMockit: Coverage source dirs: {}", sourceDirs);
72          }
73  
74          generateFileCoverageReportsWhileBuildingPackageLists();
75  
76          new StaticFiles(outputDir).copyToOutputDir(withSourceFilePages);
77          new IndexPage(outputFile, sourceDirs, sourceFilesNotFound, packageToFiles, fileToFileData).generate();
78  
79          logger.info("JMockit: Coverage report written to {}", outputFile.getParentFile().getCanonicalPath());
80      }
81  
82      private void createReportOutputDirIfNotExists() {
83          if (!outputDirCreated) {
84              File outDir = new File(outputDir);
85              outputDirCreated = outDir.mkdirs();
86          }
87      }
88  
89      @Nullable
90      private File createOutputFileForIndexPage() throws IOException {
91          File outputFile = new File(outputDir, "index.html");
92  
93          if (outputFile.exists() && !outputFile.canWrite()) {
94              logger.info("JMockit: {} is read-only; report generation canceled", outputFile.getCanonicalPath());
95              return null;
96          }
97  
98          return outputFile;
99      }
100 
101     private void generateFileCoverageReportsWhileBuildingPackageLists() throws IOException {
102         Set<Entry<String, FileCoverageData>> files = fileToFileData.entrySet();
103 
104         for (Entry<String, FileCoverageData> fileAndFileData : files) {
105             generateFileCoverageReport(fileAndFileData.getKey(), fileAndFileData.getValue());
106         }
107     }
108 
109     private void generateFileCoverageReport(@NonNull String sourceFile, @NonNull FileCoverageData fileData)
110             throws IOException {
111         if (sourceDirs != null) {
112             InputFile inputFile = InputFile.createIfFileExists(sourceDirs, sourceFile);
113 
114             if (inputFile != null) {
115                 new FileCoverageReport(outputDir, inputFile, fileData, withCallPoints).generate();
116             } else {
117                 deleteOutdatedHTMLFileIfExists(sourceFile);
118 
119                 if (sourceFilesNotFound != null) {
120                     sourceFilesNotFound.add(sourceFile);
121                 }
122             }
123         }
124 
125         addFileToPackageFileList(sourceFile);
126     }
127 
128     private void addFileToPackageFileList(@NonNull String file) {
129         int p = file.lastIndexOf('/');
130         String filePackage = p < 0 ? "" : file.substring(0, p);
131         List<String> filesInPackage = packageToFiles.get(filePackage);
132 
133         if (filesInPackage == null) {
134             filesInPackage = new ArrayList<>();
135             packageToFiles.put(filePackage, filesInPackage);
136         }
137 
138         filesInPackage.add(file.substring(p + 1));
139     }
140 
141     private void deleteOutdatedHTMLFileIfExists(@NonNull String filePath) {
142         if (!outputDirCreated) {
143             File outputFile = OutputFile.getOutputFile(outputDir, filePath);
144             // noinspection ResultOfMethodCallIgnored
145             outputFile.delete();
146         }
147     }
148 }
