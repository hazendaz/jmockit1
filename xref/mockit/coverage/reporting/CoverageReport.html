7   import java.io.File;
8   import java.io.IOException;
9   import java.util.ArrayList;
10  import java.util.Collection;
11  import java.util.HashMap;
12  import java.util.List;
13  import java.util.Map;
14  import java.util.Map.Entry;
15  import java.util.Set;
16  
17  import javax.annotation.Nonnull;
18  import javax.annotation.Nullable;
19  
20  import mockit.coverage.Configuration;
21  import mockit.coverage.data.CoverageData;
22  import mockit.coverage.data.FileCoverageData;
23  import mockit.coverage.reporting.packages.IndexPage;
24  import mockit.coverage.reporting.sourceFiles.FileCoverageReport;
25  import mockit.coverage.reporting.sourceFiles.InputFile;
26  
27  public final class CoverageReport {
28      @Nonnull
29      private final String outputDir;
30      private boolean outputDirCreated;
31      @Nullable
32      private final List<File> sourceDirs;
33      @Nonnull
34      private final Map<String, FileCoverageData> fileToFileData;
35      @Nonnull
36      private final Map<String, List<String>> packageToFiles;
37      private final boolean withCallPoints;
38      @Nullable
39      private final Collection<String> sourceFilesNotFound;
40  
41      public CoverageReport(@Nonnull String outputDir, boolean outputDirCreated, @Nullable String[] srcDirs,
42              @Nonnull CoverageData coverageData, boolean withCallPoints) {
43          this.outputDir = Configuration.getOrChooseOutputDirectory(outputDir, "coverage-report");
44          this.outputDirCreated = outputDirCreated;
45          sourceDirs = srcDirs == null ? null : new SourceFiles().buildListOfSourceDirectories(srcDirs);
46          fileToFileData = coverageData.getFileToFileData();
47          packageToFiles = new HashMap<>();
48          this.withCallPoints = withCallPoints;
49          sourceFilesNotFound = srcDirs == null ? null : new ArrayList<>();
50      }
51  
52      public void generate() throws IOException {
53          createReportOutputDirIfNotExists();
54  
55          File outputFile = createOutputFileForIndexPage();
56  
57          if (outputFile == null) {
58              return;
59          }
60  
61          boolean withSourceFilePages = sourceDirs != null;
62  
63          if (withSourceFilePages && sourceDirs.size() > 1) {
64              System.out.println("JMockit: Coverage source dirs: " + sourceDirs);
65          }
66  
67          generateFileCoverageReportsWhileBuildingPackageLists();
68  
69          new StaticFiles(outputDir).copyToOutputDir(withSourceFilePages);
70          new IndexPage(outputFile, sourceDirs, sourceFilesNotFound, packageToFiles, fileToFileData).generate();
71  
72          System.out.println("JMockit: Coverage report written to " + outputFile.getParentFile().getCanonicalPath());
73      }
74  
75      private void createReportOutputDirIfNotExists() {
76          if (!outputDirCreated) {
77              File outDir = new File(outputDir);
78              outputDirCreated = outDir.mkdirs();
79          }
80      }
81  
82      @Nullable
83      private File createOutputFileForIndexPage() throws IOException {
84          File outputFile = new File(outputDir, "index.html");
85  
86          if (outputFile.exists() && !outputFile.canWrite()) {
87              System.out
88                      .println("JMockit: " + outputFile.getCanonicalPath() + " is read-only; report generation canceled");
89              return null;
90          }
91  
92          return outputFile;
93      }
94  
95      private void generateFileCoverageReportsWhileBuildingPackageLists() throws IOException {
96          Set<Entry<String, FileCoverageData>> files = fileToFileData.entrySet();
97  
98          for (Entry<String, FileCoverageData> fileAndFileData : files) {
99              generateFileCoverageReport(fileAndFileData.getKey(), fileAndFileData.getValue());
100         }
101     }
102 
103     private void generateFileCoverageReport(@Nonnull String sourceFile, @Nonnull FileCoverageData fileData)
104             throws IOException {
105         if (sourceDirs != null) {
106             InputFile inputFile = InputFile.createIfFileExists(sourceDirs, sourceFile);
107 
108             if (inputFile != null) {
109                 new FileCoverageReport(outputDir, inputFile, fileData, withCallPoints).generate();
110             } else {
111                 deleteOutdatedHTMLFileIfExists(sourceFile);
112 
113                 if (sourceFilesNotFound != null) {
114                     sourceFilesNotFound.add(sourceFile);
115                 }
116             }
117         }
118 
119         addFileToPackageFileList(sourceFile);
120     }
121 
122     private void addFileToPackageFileList(@Nonnull String file) {
123         int p = file.lastIndexOf('/');
124         String filePackage = p < 0 ? "" : file.substring(0, p);
125         List<String> filesInPackage = packageToFiles.get(filePackage);
126 
127         if (filesInPackage == null) {
128             filesInPackage = new ArrayList<>();
129             packageToFiles.put(filePackage, filesInPackage);
130         }
131 
132         filesInPackage.add(file.substring(p + 1));
133     }
134 
135     private void deleteOutdatedHTMLFileIfExists(@Nonnull String filePath) {
136         if (!outputDirCreated) {
137             File outputFile = OutputFile.getOutputFile(outputDir, filePath);
138             // noinspection ResultOfMethodCallIgnored
139             outputFile.delete();
140         }
141     }
142 }
