17  import mockit.coverage.data.FileCoverageData;
18  import mockit.coverage.lines.LineCoverageData;
19  import mockit.coverage.lines.PerFileLineCoverage;
20  
21  import org.checkerframework.checker.index.qual.NonNegative;

23  /**
24   * Generates a XML file containing the coverage data gathered by the test run. The XML schema used is the one
25   * <a href="http://docs.sonarqube.org/display/SONAR/Generic+Test+Data">defined</a> by the SonarQube project:
26   *
27   * <pre>{@code
28   * &lt;coverage version="1">
29   *    &lt;file path="com/example/MyClass.java">
30   *       &lt;lineToCover lineNumber="5" covered="false"/>
31   *       &lt;lineToCover lineNumber="8" covered="true" branchesToCover="2" coveredBranches="1"/>
32   *    &lt;/file>
33   * &lt;/coverage>
34   * }</pre>
35   */
36  final class XmlFile {
37      @NonNull
38      private final String srcDir;
39      @NonNull
40      private final File outputFile;
41      @NonNull
42      private final CoverageData coverageData;
43  
44      XmlFile(@NonNull String outputDir, @NonNull CoverageData coverageData) {
45          // noinspection DynamicRegexReplaceableByCompiledPattern
46          String firstSrcDir = Configuration.getProperty("srcDirs", "").split("\\s*,\\s*")[0];
47          srcDir = firstSrcDir.isEmpty() ? "" : firstSrcDir + '/';
48  
49          String parentDir = Configuration.getOrChooseOutputDirectory(outputDir);
50          outputFile = new File(parentDir, "coverage.xml");
51          this.coverageData = coverageData;
52      }
53  
54      void generate() throws IOException {
55          try (Writer out = new FileWriter(outputFile, StandardCharsets.UTF_8)) {
56              out.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
57              out.write("<coverage version=\"1\">\n");
58  
59              for (Entry<String, FileCoverageData> fileAndData : coverageData.getFileToFileData().entrySet()) {
60                  String sourceFileName = fileAndData.getKey();
61                  writeOpeningXmlElementForSourceFile(out, sourceFileName);
62  
63                  PerFileLineCoverage lineInfo = fileAndData.getValue().lineCoverageInfo;
64                  writeXmlElementsForExecutableLines(out, lineInfo);
65  
66                  out.write("\t</file>\n");
67              }
68  
69              out.write("</coverage>\n");
70          }
71  
72          System.out.println("JMockit: Coverage data written to " + outputFile.getCanonicalPath());
73      }
74  
75      private void writeOpeningXmlElementForSourceFile(@NonNull Writer out, @NonNull String sourceFileName)
76              throws IOException {
77          out.write("\t<file path=\"");
78          out.write(srcDir);
79          out.write(sourceFileName);
80          out.write("\">\n");
81      }
82  
83      private static void writeXmlElementsForExecutableLines(@NonNull Writer out, @NonNull PerFileLineCoverage lineInfo)
84              throws IOException {
85          int lineCount = lineInfo.getLineCount();
86  
87          for (int lineNum = 1; lineNum <= lineCount; lineNum++) {
88              if (lineInfo.hasLineData(lineNum)) {
89                  LineCoverageData lineData = lineInfo.getLineData(lineNum);
90  
91                  out.write("\t\t<lineToCover lineNumber=\"");
92                  writeNumber(out, lineNum);
93                  out.write("\" covered=\"");
94                  out.write(Boolean.toString(lineData.isCovered()));
95  
96                  if (lineData.containsBranches()) {
97                      out.write("\" branchesToCover=\"");
98                      writeNumber(out, lineData.getNumberOfBranchingSourcesAndTargets());
99                      out.write("\" coveredBranches=\"");
100                     writeNumber(out, lineData.getNumberOfCoveredBranchingSourcesAndTargets());
101                 }
102 
103                 out.write("\"/>\n");
104             }
105         }
106     }
107 
108     private static void writeNumber(@NonNull Writer out, @NonNegative int value) throws IOException {
109         out.write(Integer.toString(value));
110     }
111 }
