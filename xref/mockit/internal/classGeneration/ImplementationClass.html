5   package mockit.internal.classGeneration;
6   
7   import java.lang.reflect.Type;
8   
9   import javax.annotation.Nonnull;
10  
11  import mockit.asm.classes.ClassReader;
12  import mockit.asm.classes.ClassVisitor;
13  import mockit.internal.ClassFile;
14  import mockit.internal.util.ClassLoad;
15  import mockit.internal.util.GeneratedClasses;
16  import mockit.internal.util.Utilities;
17  
18  /**
19   * Allows the creation of new implementation classes for interfaces and abstract classes.
20   */
21  public abstract class ImplementationClass<T> {
22      @Nonnull
23      protected final Class<?> sourceClass;
24      @Nonnull
25      protected String generatedClassName;
26  
27      protected ImplementationClass(@Nonnull Type mockedType) {
28          this(Utilities.getClassType(mockedType));
29      }
30  
31      protected ImplementationClass(@Nonnull Class<?> mockedClass) {
32          this(mockedClass, GeneratedClasses.getNameForGeneratedClass(mockedClass, null));
33      }
34  
35      protected ImplementationClass(@Nonnull Class<?> sourceClass, @Nonnull String desiredClassName) {
36          this.sourceClass = sourceClass;
37          generatedClassName = desiredClassName;
38      }
39  
40      @Nonnull
41      public final Class<T> generateClass() {
42          ClassReader classReader = ClassFile.createReaderOrGetFromCache(sourceClass);
43  
44          ClassVisitor modifier = createMethodBodyGenerator(classReader);
45          classReader.accept(modifier);
46  
47          return defineNewClass(modifier);
48      }
49  
50      @Nonnull
51      protected abstract ClassVisitor createMethodBodyGenerator(@Nonnull ClassReader cr);
52  
53      @Nonnull
54      private Class<T> defineNewClass(@Nonnull ClassVisitor modifier) {
55          final ClassLoader parentLoader = ClassLoad.getClassLoaderWithAccess(sourceClass);
56          final byte[] modifiedClassfile = modifier.toByteArray();
57  
58          try {
59              return (Class<T>) new ClassLoader(parentLoader) {
60                  @Override
61                  protected Class<?> findClass(String name) throws ClassNotFoundException {
62                      if (!name.equals(generatedClassName)) {
63                          return parentLoader.loadClass(name);
64                      }
65  
66                      return defineClass(name, modifiedClassfile, 0, modifiedClassfile.length);
67                  }
68              }.findClass(generatedClassName);
69          } catch (ClassNotFoundException e) {
70              throw new RuntimeException("Unable to define class: " + generatedClassName, e);
71          }
72      }
73  }
