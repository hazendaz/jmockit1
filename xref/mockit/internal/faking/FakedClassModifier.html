45  /**
46   * Responsible for generating all necessary bytecode in the redefined (real) class. Such code will redirect calls made
47   * on "real" methods to equivalent calls on the corresponding "fake" methods. The original code won't be executed by the
48   * running JVM until the class redefinition is undone.
49   * <p>
50   * Methods in the real class with no corresponding fake methods are unaffected.
51   * <p>
52   * Any fields (static or not) in the real class remain untouched.
53   */
54  final class FakedClassModifier extends BaseClassModifier {
55      private static final int ABSTRACT_OR_SYNTHETIC = Access.ABSTRACT + Access.SYNTHETIC;
56  
57      @Nonnull
58      private final FakeMethods fakeMethods;
59      private final boolean useClassLoadingBridgeForUpdatingFakeState;
60      @Nonnull
61      private final Class<?> fakedClass;
62      private FakeMethod fakeMethod;
63      private boolean isConstructor;
64  
65      /**
66       * Initializes the modifier for a given real/fake class pair.
67       * <p>
68       * The fake instance provided will receive calls for any instance methods defined in the fake class. Therefore, it
69       * needs to be later recovered by the modified bytecode inside the real method. To enable this, the fake instance is
70       * added to a global data structure made available through the {@link TestRun#getFake(String)} method.
71       *
72       * @param cr
73       *            the class file reader for the real class
74       * @param realClass
75       *            the class to be faked, or a base type of an implementation class to be faked
76       * @param fake
77       *            an instance of the fake class
78       * @param fakeMethods
79       *            contains the set of fake methods collected from the fake class; each fake method is identified by a
80       *            pair composed of "name" and "desc", where "name" is the method name, and "desc" is the JVM internal
81       *            description of the parameters; once the real class modification is complete this set will be empty,
82       *            unless no corresponding real method was found for any of its method identifiers
83       */
84      FakedClassModifier(@Nonnull ClassReader cr, @Nonnull Class<?> realClass, @Nonnull MockUp<?> fake,
85              @Nonnull FakeMethods fakeMethods) {
86          super(cr);
87          fakedClass = realClass;
88          this.fakeMethods = fakeMethods;
89  
90          ClassLoader classLoaderOfRealClass = realClass.getClassLoader();
91          useClassLoadingBridgeForUpdatingFakeState = ClassLoad.isClassLoaderWithNoDirectAccess(classLoaderOfRealClass);
92          inferUseOfClassLoadingBridge(classLoaderOfRealClass, fake);
93      }
94  
95      private void inferUseOfClassLoadingBridge(@Nullable ClassLoader classLoaderOfRealClass, @Nonnull Object fake) {
96          setUseClassLoadingBridge(classLoaderOfRealClass);
97  
98          if (!useClassLoadingBridge && !isPublic(fake.getClass().getModifiers())) {
99              useClassLoadingBridge = true;
100         }
101     }

103     @Override
104     public MethodVisitor visitMethod(int access, @Nonnull String name, @Nonnull String desc, @Nullable String signature,
105             @Nullable String[] exceptions) {
106         if ((access & ABSTRACT_OR_SYNTHETIC) != 0) {
107             if (isAbstract(access)) {
108                 // Marks a matching fake method (if any) as having the corresponding faked method.
109                 fakeMethods.findMethod(access, name, desc, signature);
110             }
111 
112             return cw.visitMethod(access, name, desc, signature, exceptions);
113         }
114 
115         isConstructor = "<init>".equals(name);
116 
117         if (isConstructor && isFakedSuperclass() || !hasFake(access, name, desc, signature)) {
118             return cw.visitMethod(access, name, desc, signature, exceptions);
119         }
120 
121         if (isPrivate(access)) {
122             String kindOfMember = isConstructor ? "constructor " : "method ";
123             String privateMemberDesc = fakedClass.getSimpleName() + '#' + name + desc;
124             throw new IllegalArgumentException(
125                     "Unsupported fake for private " + kindOfMember + privateMemberDesc + " found");
126         }
127 
128         startModifiedMethodVersion(access, name, desc, signature, exceptions);
129 
130         if (isNative(methodAccess)) {
131             generateCodeForInterceptedNativeMethod();
132             return methodAnnotationsVisitor;
133         }
134 
135         return copyOriginalImplementationWithInjectedInterceptionCode();
136     }
137 
138     private boolean hasFake(int access, @Nonnull String name, @Nonnull String desc, @Nullable String signature) {
139         String fakeName = getCorrespondingFakeName(name);
140         fakeMethod = fakeMethods.findMethod(access, fakeName, desc, signature);
141         return fakeMethod != null;
142     }
143 
144     @Nonnull
145     private static String getCorrespondingFakeName(@Nonnull String name) {
146         if ("<init>".equals(name)) {
147             return "$init";
148         }
149 
150         if ("<clinit>".equals(name)) {
151             return "$clinit";
152         }
153 
154         return name;
155     }
156 
157     private boolean isFakedSuperclass() {
158         return fakedClass != fakeMethods.getRealClass();
159     }
160 
161     private void generateCodeForInterceptedNativeMethod() {
162         generateCallToUpdateFakeState();
163         generateCallToFakeMethod();
164         generateMethodReturn();
165         mw.visitMaxStack(1); // dummy value, real one is calculated by ASM
166     }
167 
168     @Override
169     protected void generateInterceptionCode() {
170         Label startOfRealImplementation = null;
171 
172         if (!isStatic(methodAccess) && !isConstructor && isFakedSuperclass()) {
173             Class<?> targetClass = fakeMethods.getRealClass();
174 
175             if (fakedClass.getClassLoader() == targetClass.getClassLoader()) {
176                 startOfRealImplementation = new Label();
177                 mw.visitVarInsn(ALOAD, 0);
178                 mw.visitTypeInsn(INSTANCEOF, JavaType.getInternalName(targetClass));
179                 mw.visitJumpInsn(IFEQ, startOfRealImplementation);
180             }
181         }
182 
183         generateCallToUpdateFakeState();
184 
185         if (isConstructor) {
186             generateConditionalCallForFakedConstructor();
187         } else {
188             generateConditionalCallForFakedMethod(startOfRealImplementation);
189         }
190     }
191 
192     private void generateCallToUpdateFakeState() {
193         if (useClassLoadingBridgeForUpdatingFakeState) {
194             generateCallToControlMethodThroughClassLoadingBridge();
195             mw.visitTypeInsn(CHECKCAST, "java/lang/Boolean");
196             mw.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Boolean", "booleanValue", "()Z", false);
197         } else {
198             mw.visitLdcInsn(fakeMethods.getFakeClassInternalName());
199             mw.visitIntInsn(SIPUSH, fakeMethod.getIndexForFakeState());
200             mw.visitMethodInsn(INVOKESTATIC, "mockit/internal/state/TestRun", "updateFakeState",
201                     "(Ljava/lang/String;I)Z", false);
202         }
203     }
204 
205     private void generateCallToControlMethodThroughClassLoadingBridge() {
206         generateCodeToObtainInstanceOfClassLoadingBridge(FakeBridge.MB);
207 
208         // First and second "invoke" arguments:
209         generateCodeToPassThisOrNullIfStaticMethod();
210         mw.visitInsn(ACONST_NULL);
211 
212         // Create array for call arguments (third "invoke" argument):
213         generateCodeToCreateArrayOfObject(2);
214 
215         int i = 0;
216         generateCodeToFillArrayElement(i, fakeMethods.getFakeClassInternalName());
217         i++;
218         generateCodeToFillArrayElement(i, fakeMethod.getIndexForFakeState());
219 
220         generateCallToInvocationHandler();
221     }

223     private void generateConditionalCallForFakedMethod(@Nullable Label startOfRealImplementation) {
224         if (startOfRealImplementation == null) {
225             // noinspection AssignmentToMethodParameter
226             startOfRealImplementation = new Label();
227         }
228 
229         mw.visitJumpInsn(IFEQ, startOfRealImplementation);
230         generateCallToFakeMethod();
231         generateMethodReturn();
232         mw.visitLabel(startOfRealImplementation);
233     }
234 
235     private void generateConditionalCallForFakedConstructor() {
236         generateCallToFakeMethod();
237 
238         int jumpInsnOpcode;
239 
240         if (shouldUseClassLoadingBridge()) {
241             mw.visitLdcInsn(VOID_TYPE);
242             jumpInsnOpcode = IF_ACMPEQ;
243         } else {
244             jumpInsnOpcode = fakeMethod.hasInvocationParameter() ? IFNE : IFEQ;
245         }
246 
247         Label startOfRealImplementation = new Label();
248         mw.visitJumpInsn(jumpInsnOpcode, startOfRealImplementation);
249         mw.visitInsn(RETURN);
250         mw.visitLabel(startOfRealImplementation);
251     }
252 
253     private void generateCallToFakeMethod() {
254         if (shouldUseClassLoadingBridge()) {
255             generateCallToFakeMethodThroughClassLoadingBridge();
256         } else {
257             generateDirectCallToFakeMethod();
258         }
259     }
260 
261     private boolean shouldUseClassLoadingBridge() {
262         return useClassLoadingBridge || !fakeMethod.isPublic();
263     }
264 
265     private void generateCallToFakeMethodThroughClassLoadingBridge() {
266         generateCodeToObtainInstanceOfClassLoadingBridge(FakeMethodBridge.MB);
267 
268         // First and second "invoke" arguments:
269         boolean isStatic = generateCodeToPassThisOrNullIfStaticMethod();
270         mw.visitInsn(ACONST_NULL);
271 
272         // Create array for call arguments (third "invoke" argument):
273         JavaType[] argTypes = JavaType.getArgumentTypes(methodDesc);
274         generateCodeToCreateArrayOfObject(6 + argTypes.length);
275 
276         int i = 0;
277         generateCodeToFillArrayElement(i, fakeMethods.getFakeClassInternalName());
278         i++;
279         generateCodeToFillArrayElement(i, classDesc);
280         i++;
281         generateCodeToFillArrayElement(i, methodAccess);
282         i++;
283 
284         if (fakeMethod.hasInvocationParameterOnly()) {
285             generateCodeToFillArrayElement(i, methodName);
286             i++;
287             generateCodeToFillArrayElement(i, methodDesc);
288         } else {
289             generateCodeToFillArrayElement(i, fakeMethod.name);
290             i++;
291             generateCodeToFillArrayElement(i, fakeMethod.desc);
292         }
293         i++;
294 
295         generateCodeToFillArrayElement(i, fakeMethod.getIndexForFakeState());
296         i++;
297 
298         generateCodeToFillArrayWithParameterValues(argTypes, i, isStatic ? 0 : 1);
299         generateCallToInvocationHandler();
300     }

302     private void generateDirectCallToFakeMethod() {
303         String fakeClassDesc = fakeMethods.getFakeClassInternalName();
304         int invokeOpcode;
305 
306         if (fakeMethod.isStatic()) {
307             invokeOpcode = INVOKESTATIC;
308         } else {
309             generateCodeToObtainFakeInstance(fakeClassDesc);
310             invokeOpcode = INVOKEVIRTUAL;
311         }
312 
313         boolean canProceedIntoConstructor = generateArgumentsForFakeMethodInvocation();
314         mw.visitMethodInsn(invokeOpcode, fakeClassDesc, fakeMethod.name, fakeMethod.desc, false);
315 
316         if (canProceedIntoConstructor) {
317             mw.visitMethodInsn(INVOKEVIRTUAL, "mockit/internal/faking/FakeInvocation", "shouldProceedIntoConstructor",
318                     "()Z", false);
319         }
320     }
321 
322     private void generateCodeToObtainFakeInstance(@Nonnull String fakeClassDesc) {
323         mw.visitLdcInsn(fakeClassDesc);
324         mw.visitMethodInsn(INVOKESTATIC, "mockit/internal/state/TestRun", "getFake",
325                 "(Ljava/lang/String;)Ljava/lang/Object;", false);
326         mw.visitTypeInsn(CHECKCAST, fakeClassDesc);
327     }

329     private boolean generateArgumentsForFakeMethodInvocation() {
330         String fakedDesc = fakeMethod.hasInvocationParameterOnly() ? methodDesc
331                 : fakeMethod.fakeDescWithoutInvocationParameter;
332         JavaType[] argTypes = JavaType.getArgumentTypes(fakedDesc);
333         int varIndex = isStatic(methodAccess) ? 0 : 1;
334         boolean canProceedIntoConstructor = false;
335 
336         if (fakeMethod.hasInvocationParameter()) {
337             generateCallToCreateNewFakeInvocation(argTypes, varIndex);
338 
339             // When invoking a constructor, the invocation object will need to be consulted for proceeding:
340             if (isConstructor) {
341                 mw.visitInsn(fakeMethod.isStatic() ? DUP : DUP_X1);
342                 canProceedIntoConstructor = true;
343             }
344         }
345 
346         if (!fakeMethod.hasInvocationParameterOnly()) {
347             passArgumentsForFakeMethodCall(argTypes, varIndex);
348         }
349 
350         return canProceedIntoConstructor;
351     }
352 
353     private void generateCallToCreateNewFakeInvocation(@Nonnull JavaType[] argTypes,
354             @Nonnegative int initialParameterIndex) {
355         generateCodeToPassThisOrNullIfStaticMethod();
356 
357         int argCount = argTypes.length;
358 
359         if (argCount == 0) {
360             mw.visitInsn(ACONST_NULL);
361         } else {
362             generateCodeToCreateArrayOfObject(argCount);
363             generateCodeToFillArrayWithParameterValues(argTypes, 0, initialParameterIndex);
364         }
365 
366         mw.visitLdcInsn(fakeMethods.getFakeClassInternalName());
367         mw.visitIntInsn(SIPUSH, fakeMethod.getIndexForFakeState());
368         mw.visitLdcInsn(classDesc);
369         mw.visitLdcInsn(methodName);
370         mw.visitLdcInsn(methodDesc);
371 
372         mw.visitMethodInsn(INVOKESTATIC, "mockit/internal/faking/FakeInvocation", "create",
373                 "(Ljava/lang/Object;[Ljava/lang/Object;Ljava/lang/String;ILjava/lang/String;Ljava/lang/String;Ljava/lang/String;)"
374                         + "Lmockit/internal/faking/FakeInvocation;",
375                 false);
376     }
377 
378     private void passArgumentsForFakeMethodCall(@Nonnull JavaType[] argTypes, @Nonnegative int varIndex) {
379         boolean forGenericMethod = fakeMethod.isForGenericMethod();
380 
381         for (JavaType argType : argTypes) {
382             int opcode = argType.getOpcode(ILOAD);
383             mw.visitVarInsn(opcode, varIndex);
384 
385             if (forGenericMethod && argType instanceof ReferenceType) {
386                 String typeDesc = ((ReferenceType) argType).getInternalName();
387                 mw.visitTypeInsn(CHECKCAST, typeDesc);
388             }
389 
390             varIndex += argType.getSize();
391         }
392     }
393 
394     private void generateMethodReturn() {
395         if (shouldUseClassLoadingBridge() || fakeMethod.isAdvice) {
396             generateReturnWithObjectAtTopOfTheStack(methodDesc);
397         } else {
398             JavaType returnType = JavaType.getReturnType(methodDesc);
399             mw.visitInsn(returnType.getOpcode(IRETURN));
400         }
401     }
