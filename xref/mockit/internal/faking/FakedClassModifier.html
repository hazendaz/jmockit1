48  /**
49   * Responsible for generating all necessary bytecode in the redefined (real) class. Such code will redirect calls made
50   * on "real" methods to equivalent calls on the corresponding "fake" methods. The original code won't be executed by the
51   * running JVM until the class redefinition is undone.
52   * <p>
53   * Methods in the real class with no corresponding fake methods are unaffected.
54   * <p>
55   * Any fields (static or not) in the real class remain untouched.
56   */
57  final class FakedClassModifier extends BaseClassModifier {
58      private static final int ABSTRACT_OR_SYNTHETIC = Access.ABSTRACT + Access.SYNTHETIC;
59  
60      @NonNull
61      private final FakeMethods fakeMethods;
62      private final boolean useClassLoadingBridgeForUpdatingFakeState;
63      @NonNull
64      private final Class<?> fakedClass;
65      private FakeMethod fakeMethod;
66      private boolean isConstructor;
67  
68      /**
69       * Initializes the modifier for a given real/fake class pair.
70       * <p>
71       * The fake instance provided will receive calls for any instance methods defined in the fake class. Therefore, it
72       * needs to be later recovered by the modified bytecode inside the real method. To enable this, the fake instance is
73       * added to a global data structure made available through the {@link TestRun#getFake(String, Object)} method.
74       *
75       * @param cr
76       *            the class file reader for the real class
77       * @param realClass
78       *            the class to be faked, or a base type of an implementation class to be faked
79       * @param fake
80       *            an instance of the fake class
81       * @param fakeMethods
82       *            contains the set of fake methods collected from the fake class; each fake method is identified by a
83       *            pair composed of "name" and "desc", where "name" is the method name, and "desc" is the JVM internal
84       *            description of the parameters; once the real class modification is complete this set will be empty,
85       *            unless no corresponding real method was found for any of its method identifiers
86       */
87      FakedClassModifier(@NonNull ClassReader cr, @NonNull Class<?> realClass, @NonNull MockUp<?> fake,
88              @NonNull FakeMethods fakeMethods) {
89          super(cr);
90          fakedClass = realClass;
91          this.fakeMethods = fakeMethods;
92  
93          ClassLoader classLoaderOfRealClass = realClass.getClassLoader();
94          useClassLoadingBridgeForUpdatingFakeState = ClassLoad.isClassLoaderWithNoDirectAccess(classLoaderOfRealClass);
95          inferUseOfClassLoadingBridge(classLoaderOfRealClass, fake);
96      }
97  
98      private void inferUseOfClassLoadingBridge(@Nullable ClassLoader classLoaderOfRealClass, @NonNull Object fake) {
99          setUseClassLoadingBridge(classLoaderOfRealClass);
100 
101         if (!useClassLoadingBridge && !isPublic(fake.getClass().getModifiers())) {
102             useClassLoadingBridge = true;
103         }
104     }

106     @Override
107     public MethodVisitor visitMethod(int access, @NonNull String name, @NonNull String desc, @Nullable String signature,
108             @Nullable String[] exceptions) {
109         if ((access & ABSTRACT_OR_SYNTHETIC) != 0) {
110             if (isAbstract(access)) {
111                 // Marks a matching fake method (if any) as having the corresponding faked method.
112                 fakeMethods.findMethod(access, name, desc, signature);
113             }
114 
115             return cw.visitMethod(access, name, desc, signature, exceptions);
116         }
117 
118         isConstructor = "<init>".equals(name);
119 
120         if (isConstructor && isFakedSuperclass() || !hasFake(access, name, desc, signature)) {
121             return cw.visitMethod(access, name, desc, signature, exceptions);
122         }
123 
124         startModifiedMethodVersion(access, name, desc, signature, exceptions);
125 
126         if (isNative(methodAccess)) {
127             generateCodeForInterceptedNativeMethod();
128             return methodAnnotationsVisitor;
129         }
130 
131         return copyOriginalImplementationWithInjectedInterceptionCode();
132     }
133 
134     private boolean hasFake(int access, @NonNull String name, @NonNull String desc, @Nullable String signature) {
135         String fakeName = getCorrespondingFakeName(name);
136         fakeMethod = fakeMethods.findMethod(access, fakeName, desc, signature);
137         return fakeMethod != null;
138     }
139 
140     @NonNull
141     private static String getCorrespondingFakeName(@NonNull String name) {
142         if ("<init>".equals(name)) {
143             return "$init";
144         }
145 
146         if ("<clinit>".equals(name)) {
147             return "$clinit";
148         }
149 
150         return name;
151     }
152 
153     private boolean isFakedSuperclass() {
154         return fakedClass != fakeMethods.getRealClass();
155     }

157     private void generateCodeForInterceptedNativeMethod() {
158         generateCallToUpdateFakeState();
159         generateCallToFakeMethod();
160         generateMethodReturn();
161         mw.visitMaxStack(1); // dummy value, real one is calculated by ASM
162     }

164     @Override
165     protected void generateInterceptionCode() {
166         Label startOfRealImplementation = null;
167 
168         if (!isStatic(methodAccess) && !isConstructor && isFakedSuperclass()) {
169             Class<?> targetClass = fakeMethods.getRealClass();
170 
171             if (fakedClass.getClassLoader() == targetClass.getClassLoader()) {
172                 startOfRealImplementation = new Label();
173                 mw.visitVarInsn(ALOAD, 0);
174                 mw.visitTypeInsn(INSTANCEOF, JavaType.getInternalName(targetClass));
175                 mw.visitJumpInsn(IFEQ, startOfRealImplementation);
176             }
177         }
178 
179         generateCallToUpdateFakeState();
180 
181         if (isConstructor) {
182             generateConditionalCallForFakedConstructor();
183         } else {
184             generateConditionalCallForFakedMethod(startOfRealImplementation);
185         }
186     }
187 
188     private void generateCallToUpdateFakeState() {
189         if (useClassLoadingBridgeForUpdatingFakeState) {
190             generateCallToControlMethodThroughClassLoadingBridge();
191             mw.visitTypeInsn(CHECKCAST, "java/lang/Boolean");
192             mw.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Boolean", "booleanValue", "()Z", false);
193         } else {
194             mw.visitLdcInsn(fakeMethods.getFakeClassInternalName());
195             generateCodeToPassThisOrNullIfStaticMethod();
196             mw.visitIntInsn(SIPUSH, fakeMethod.getIndexForFakeState());
197             mw.visitMethodInsn(INVOKESTATIC, "mockit/internal/state/TestRun", "updateFakeState",
198                     "(Ljava/lang/String;Ljava/lang/Object;I)Z", false);
199         }
200     }
201 
202     private void generateCallToControlMethodThroughClassLoadingBridge() {
203         generateCodeToObtainInstanceOfClassLoadingBridge(FakeBridge.MB);
204 
205         // First and second "invoke" arguments:
206         generateCodeToPassThisOrNullIfStaticMethod();
207         mw.visitInsn(ACONST_NULL);
208 
209         // Create array for call arguments (third "invoke" argument):
210         generateCodeToCreateArrayOfObject(2);
211 
212         int i = 0;
213         generateCodeToFillArrayElement(i, fakeMethods.getFakeClassInternalName());
214         i++;
215         generateCodeToFillArrayElement(i, fakeMethod.getIndexForFakeState());
216 
217         generateCallToInvocationHandler();
218     }

220     private void generateConditionalCallForFakedMethod(@Nullable Label startOfRealImplementation) {
221         if (startOfRealImplementation == null) {
222             // noinspection AssignmentToMethodParameter
223             startOfRealImplementation = new Label();
224         }
225 
226         mw.visitJumpInsn(IFEQ, startOfRealImplementation);
227         generateCallToFakeMethod();
228         generateMethodReturn();
229         mw.visitLabel(startOfRealImplementation);
230     }
231 
232     private void generateConditionalCallForFakedConstructor() {
233         generateCallToFakeMethod();
234 
235         int jumpInsnOpcode;
236 
237         if (shouldUseClassLoadingBridge()) {
238             mw.visitLdcInsn(VOID_TYPE);
239             jumpInsnOpcode = IF_ACMPEQ;
240         } else {
241             jumpInsnOpcode = fakeMethod.hasInvocationParameter() ? IFNE : IFEQ;
242         }
243 
244         Label startOfRealImplementation = new Label();
245         mw.visitJumpInsn(jumpInsnOpcode, startOfRealImplementation);
246         mw.visitInsn(RETURN);
247         mw.visitLabel(startOfRealImplementation);
248     }

250     /**
251      * Generates an early fake interception check for constructors that use {@code this()} delegation. The check runs
252      * before any of the original argument-creation bytecode so that exceptions thrown while building the arguments to
253      * {@code this()} cannot escape when the fake is active.
254      * <p>
255      * When the fake is active the method:
256      * <ol>
257      * <li>Initialises {@code this} via the direct superclass's no-argument constructor (which must exist).</li>
258      * <li>Calls the fake {@code $init} method with the original constructor's parameters.</li>
259      * <li>Returns immediately, skipping the {@code this()} delegation entirely.</li>
260      * </ol>
261      * When the fake is not active (or the superclass has no no-argument constructor) execution falls through to
262      * {@code originalCodeLabel} and the original constructor body runs normally.
263      */
264     @Override
265     protected void generateEarlyInterceptionCodeForThisDelegationConstructor(@NonNull Label originalCodeLabel) {
266         if (!superClassHasNoArgConstructor()) {
267             // Cannot safely initialise 'this' – fall back to post-this() interception only
268             mw.visitLabel(originalCodeLabel);
269             return;
270         }
271 
272         // Check whether the fake is active, passing null for 'this' (which is still uninitialized here)
273         generateCallToUpdateFakeStateWithNullInstance(); // leaves Z on stack
274 
275         // If not active, jump to the original constructor body
276         mw.visitJumpInsn(IFEQ, originalCodeLabel);
277 
278         // --- Fake-active path ---
279         // Initialise 'this' via the direct superclass's no-argument constructor so that RETURN is legal
280         mw.visitVarInsn(ALOAD, 0);
281         mw.visitMethodInsn(INVOKESPECIAL, superClassName, "<init>", "()V", false);
282 
283         // Call the fake $init method (ALOAD_0 is now a properly-initialised reference)
284         generateCallToFakeMethod();
285 
286         // If the fake has an Invocation parameter, shouldProceedIntoConstructor() left a Z on the stack.
287         // Calling the fake with Invocation IS supported (e.g., inv.getInvokedInstance(), inv.getInvocationCount()),
288         // but inv.proceed() is not: this path has already initialised 'this' via the superclass no-arg constructor
289         // rather than via the original this() chain, so proceeding would require re-running the original
290         // argument-creation code. We therefore discard the shouldProceedIntoConstructor() result and always return.
291         if (fakeMethod.hasInvocationParameter()) {
292             mw.visitInsn(POP);
293         }
294 
295         mw.visitInsn(RETURN);
296 
297         // --- Original-code path ---
298         mw.visitLabel(originalCodeLabel);
299     }

301     /**
302      * Returns {@code true} when the direct superclass of the faked class declares a no-argument constructor. The result
303      * determines whether an early fake check can be inserted for {@code this()}-delegation constructors.
304      */
305     private boolean superClassHasNoArgConstructor() {
306         Class<?> superClass = fakedClass.getSuperclass();
307 
308         // superClass is null only for java.lang.Object itself, which cannot realistically be faked
309         // with a constructor mock; treat it as "no no-arg constructor" to fall back safely.
310         if (superClass == null) {
311             return false;
312         }
313 
314         try {
315             superClass.getDeclaredConstructor();
316             return true;
317         } catch (NoSuchMethodException e) {
318             return false;
319         }
320     }

322     /**
323      * Like {@link #generateCallToUpdateFakeState()}, but always passes {@code null} as the mocked-instance argument.
324      * This is used at the very start of a constructor (before {@code super()}/{@code this()} has been called) when
325      * {@code this} is still an {@code uninitializedThis} reference and cannot legally be passed to a static method.
326      */
327     private void generateCallToUpdateFakeStateWithNullInstance() {
328         if (useClassLoadingBridgeForUpdatingFakeState) {
329             generateCodeToObtainInstanceOfClassLoadingBridge(FakeBridge.MB);
330             mw.visitInsn(ACONST_NULL); // null for 'this' (uninitialized)
331             mw.visitInsn(ACONST_NULL);
332             generateCodeToCreateArrayOfObject(2);
333             int i = 0;
334             generateCodeToFillArrayElement(i, fakeMethods.getFakeClassInternalName());
335             i++;
336             generateCodeToFillArrayElement(i, fakeMethod.getIndexForFakeState());
337             generateCallToInvocationHandler();
338             mw.visitTypeInsn(CHECKCAST, "java/lang/Boolean");
339             mw.visitMethodInsn(INVOKEVIRTUAL, "java/lang/Boolean", "booleanValue", "()Z", false);
340         } else {
341             mw.visitLdcInsn(fakeMethods.getFakeClassInternalName());
342             mw.visitInsn(ACONST_NULL); // null for 'this' (uninitialized)
343             mw.visitIntInsn(SIPUSH, fakeMethod.getIndexForFakeState());
344             mw.visitMethodInsn(INVOKESTATIC, "mockit/internal/state/TestRun", "updateFakeState",
345                     "(Ljava/lang/String;Ljava/lang/Object;I)Z", false);
346         }
347     }

349     private void generateCallToFakeMethod() {
350         if (shouldUseClassLoadingBridge()) {
351             generateCallToFakeMethodThroughClassLoadingBridge();
352         } else {
353             generateDirectCallToFakeMethod();
354         }
355     }

357     private boolean shouldUseClassLoadingBridge() {
358         return useClassLoadingBridge || !fakeMethod.isPublic();
359     }
360 
361     private void generateCallToFakeMethodThroughClassLoadingBridge() {
362         generateCodeToObtainInstanceOfClassLoadingBridge(FakeMethodBridge.MB);
363 
364         // First and second "invoke" arguments:
365         boolean isStatic = generateCodeToPassThisOrNullIfStaticMethod();
366         mw.visitInsn(ACONST_NULL);
367 
368         // Create array for call arguments (third "invoke" argument):
369         JavaType[] argTypes = JavaType.getArgumentTypes(methodDesc);
370         generateCodeToCreateArrayOfObject(6 + argTypes.length);
371 
372         int i = 0;
373         generateCodeToFillArrayElement(i, fakeMethods.getFakeClassInternalName());
374         i++;
375         generateCodeToFillArrayElement(i, classDesc);
376         i++;
377         generateCodeToFillArrayElement(i, methodAccess);
378         i++;
379 
380         if (fakeMethod.hasInvocationParameterOnly()) {
381             generateCodeToFillArrayElement(i, methodName);
382             i++;
383             generateCodeToFillArrayElement(i, methodDesc);
384         } else {
385             generateCodeToFillArrayElement(i, fakeMethod.name);
386             i++;
387             generateCodeToFillArrayElement(i, fakeMethod.desc);
388         }
389         i++;
390 
391         generateCodeToFillArrayElement(i, fakeMethod.getIndexForFakeState());
392         i++;
393 
394         generateCodeToFillArrayWithParameterValues(argTypes, i, isStatic ? 0 : 1);
395         generateCallToInvocationHandler();
396     }

398     private void generateDirectCallToFakeMethod() {
399         String fakeClassDesc = fakeMethods.getFakeClassInternalName();
400         int invokeOpcode;
401 
402         if (fakeMethod.isStatic()) {
403             invokeOpcode = INVOKESTATIC;
404         } else {
405             generateCodeToObtainFakeInstance(fakeClassDesc);
406             invokeOpcode = INVOKEVIRTUAL;
407         }
408 
409         boolean canProceedIntoConstructor = generateArgumentsForFakeMethodInvocation();
410         mw.visitMethodInsn(invokeOpcode, fakeClassDesc, fakeMethod.name, fakeMethod.desc, false);
411 
412         if (canProceedIntoConstructor) {
413             mw.visitMethodInsn(INVOKEVIRTUAL, "mockit/internal/faking/FakeInvocation", "shouldProceedIntoConstructor",
414                     "()Z", false);
415         }
416     }
417 
418     private void generateCodeToObtainFakeInstance(@NonNull String fakeClassDesc) {
419         mw.visitLdcInsn(fakeClassDesc);
420         generateCodeToPassThisOrNullIfStaticMethod();
421         mw.visitMethodInsn(INVOKESTATIC, "mockit/internal/state/TestRun", "getFake",
422                 "(Ljava/lang/String;Ljava/lang/Object;)Ljava/lang/Object;", false);
423         mw.visitTypeInsn(CHECKCAST, fakeClassDesc);
424     }

426     private boolean generateArgumentsForFakeMethodInvocation() {
427         String fakedDesc = fakeMethod.hasInvocationParameterOnly() ? methodDesc
428                 : fakeMethod.fakeDescWithoutInvocationParameter;
429         JavaType[] argTypes = JavaType.getArgumentTypes(fakedDesc);
430         int varIndex = isStatic(methodAccess) ? 0 : 1;
431         boolean canProceedIntoConstructor = false;
432 
433         if (fakeMethod.hasInvocationParameter()) {
434             generateCallToCreateNewFakeInvocation(argTypes, varIndex);
435 
436             // When invoking a constructor, the invocation object will need to be consulted for proceeding:
437             if (isConstructor) {
438                 mw.visitInsn(fakeMethod.isStatic() ? DUP : DUP_X1);
439                 canProceedIntoConstructor = true;
440             }
441         }
442 
443         if (!fakeMethod.hasInvocationParameterOnly()) {
444             passArgumentsForFakeMethodCall(argTypes, varIndex);
445         }
446 
447         return canProceedIntoConstructor;
448     }
449 
450     private void generateCallToCreateNewFakeInvocation(@NonNull JavaType[] argTypes,
451             @NonNegative int initialParameterIndex) {
452         generateCodeToPassThisOrNullIfStaticMethod();
453 
454         int argCount = argTypes.length;
455 
456         if (argCount == 0) {
457             mw.visitInsn(ACONST_NULL);
458         } else {
459             generateCodeToCreateArrayOfObject(argCount);
460             generateCodeToFillArrayWithParameterValues(argTypes, 0, initialParameterIndex);
461         }
462 
463         mw.visitLdcInsn(fakeMethods.getFakeClassInternalName());
464         mw.visitIntInsn(SIPUSH, fakeMethod.getIndexForFakeState());
465         mw.visitLdcInsn(classDesc);
466         mw.visitLdcInsn(methodName);
467         mw.visitLdcInsn(methodDesc);
468 
469         mw.visitMethodInsn(INVOKESTATIC, "mockit/internal/faking/FakeInvocation", "create",
470                 "(Ljava/lang/Object;[Ljava/lang/Object;Ljava/lang/String;ILjava/lang/String;Ljava/lang/String;Ljava/lang/String;)"
471                         + "Lmockit/internal/faking/FakeInvocation;",
472                 false);
473     }
474 
475     private void passArgumentsForFakeMethodCall(@NonNull JavaType[] argTypes, @NonNegative int varIndex) {
476         boolean forGenericMethod = fakeMethod.isForGenericMethod();
477 
478         for (JavaType argType : argTypes) {
479             int opcode = argType.getOpcode(ILOAD);
480             mw.visitVarInsn(opcode, varIndex);
481 
482             if (forGenericMethod && argType instanceof ReferenceType) {
483                 String typeDesc = ((ReferenceType) argType).getInternalName();
484                 mw.visitTypeInsn(CHECKCAST, typeDesc);
485             }
486 
487             varIndex += argType.getSize();
488         }
489     }
490 
491     private void generateMethodReturn() {
492         if (shouldUseClassLoadingBridge() || fakeMethod.isAdvice) {
493             generateReturnWithObjectAtTopOfTheStack(methodDesc);
494         } else {
495             JavaType returnType = JavaType.getReturnType(methodDesc);
496             mw.visitInsn(returnType.getOpcode(IRETURN));
497         }
498     }
