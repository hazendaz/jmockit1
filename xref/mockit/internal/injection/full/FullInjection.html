38  /**
39   * Responsible for recursive injection of dependencies into a <code>@Tested(fullyInitialized = true)</code> object.
40   */
41  public final class FullInjection {
42      private static final int INVALID_TYPES = Access.ABSTRACT + Access.ANNOTATION + Access.ENUM;
43  
44      @Nonnull
45      private final InjectionState injectionState;
46      @Nonnull
47      private final String testedClassName;
48      @Nonnull
49      private final String testedName;
50      @Nullable
51      private final ServletDependencies servletDependencies;
52      @Nullable
53      private final JPADependencies jpaDependencies;
54      @Nullable
55      private Class<?> dependencyClass;
56      @Nullable
57      private InjectionProvider parentInjectionProvider;
58  
59      public FullInjection(@Nonnull InjectionState injectionState, @Nonnull Class<?> testedClass,
60              @Nonnull String testedName) {
61          this.injectionState = injectionState;
62          testedClassName = testedClass.getSimpleName();
63          this.testedName = testedName;
64          servletDependencies = SERVLET_CLASS == null ? null : new ServletDependencies(injectionState);
65          jpaDependencies = PERSISTENCE_UNIT_CLASS == null ? null : new JPADependencies(injectionState);
66      }

68      @Nullable
69      public Object createOrReuseInstance(@Nonnull TestedClass testedClass, @Nonnull Injector injector,
70              @Nullable InjectionProvider injectionProvider, @Nullable String qualifiedName) {
71          setInjectionProvider(injectionProvider);
72  
73          InjectionPoint injectionPoint = getInjectionPoint(testedClass, injectionProvider, qualifiedName);
74          Object dependency = injectionState.getInstantiatedDependency(testedClass, injectionPoint);
75  
76          if (dependency != null) {
77              return dependency;
78          }
79  
80          Class<?> typeToInject = dependencyClass;
81  
82          if (typeToInject == Logger.class) {
83              return createLogger(testedClass);
84          }
85  
86          if (typeToInject == null || !isInstantiableType(typeToInject)) {
87              return null;
88          }
89  
90          return createInstance(testedClass, injector, injectionProvider, injectionPoint);
91      }
92  
93      public void setInjectionProvider(@Nullable InjectionProvider injectionProvider) {
94          if (injectionProvider != null) {
95              injectionProvider.parent = parentInjectionProvider;
96          }
97  
98          parentInjectionProvider = injectionProvider;
99      }

101     @Nonnull
102     private InjectionPoint getInjectionPoint(@Nonnull TestedClass testedClass,
103             @Nullable InjectionProvider injectionProvider, @Nullable String qualifiedName) {
104         if (injectionProvider == null) {
105             dependencyClass = testedClass.targetClass;
106             return new InjectionPoint(dependencyClass, qualifiedName, true);
107         }
108 
109         Type dependencyType = injectionProvider.getDeclaredType();
110 
111         if (dependencyType instanceof TypeVariable<?>) {
112             dependencyType = testedClass.reflection.resolveTypeVariable((TypeVariable<?>) dependencyType);
113             dependencyClass = getClassType(dependencyType);
114         } else {
115             dependencyClass = injectionProvider.getClassOfDeclaredType();
116         }
117 
118         if (qualifiedName != null && !qualifiedName.isEmpty()) {
119             return new InjectionPoint(dependencyClass, qualifiedName, true);
120         }
121 
122         if (jpaDependencies != null && JPADependencies.isApplicable(dependencyClass)) {
123             for (Annotation annotation : injectionProvider.getAnnotations()) {
124                 InjectionPoint injectionPoint = jpaDependencies.getInjectionPointIfAvailable(annotation);
125 
126                 if (injectionPoint != null) {
127                     return injectionPoint;
128                 }
129             }
130         }
131 
132         return new InjectionPoint(dependencyType, injectionProvider.getName(), false);
133     }

135     @Nonnull
136     private static Object createLogger(@Nonnull TestedClass testedClass) {
137         TestedClass testedClassWithLogger = testedClass.parent;
138         assert testedClassWithLogger != null;
139         return Logger.getLogger(testedClassWithLogger.nameOfTestedClass);
140     }

142     public static boolean isInstantiableType(@Nonnull Class<?> type) {
143         if (type.isPrimitive() || type.isArray() || type.isAnnotation()) {
144             return false;
145         }
146 
147         if (type.isInterface()) {
148             return true;
149         }
150 
151         int typeModifiers = type.getModifiers();
152 
153         if ((typeModifiers & INVALID_TYPES) != 0 || !isStatic(typeModifiers) && type.isMemberClass()) {
154             return false;
155         }
156 
157         return type.getClassLoader() != null;
158     }

160     @Nullable
161     private Object createInstance(@Nonnull TestedClass testedClass, @Nonnull Injector injector,
162             @Nullable InjectionProvider injectionProvider, @Nonnull InjectionPoint injectionPoint) {
163         @SuppressWarnings("ConstantConditions")
164         @Nonnull
165         Class<?> typeToInject = dependencyClass;
166         Object dependency = null;
167 
168         if (typeToInject.isInterface()) {
169             dependency = createInstanceOfSupportedInterfaceIfApplicable(testedClass, typeToInject, injectionPoint,
170                     injectionProvider);
171 
172             if (dependency == null && typeToInject.getClassLoader() != null) {
173                 Class<?> resolvedType = injectionState.resolveInterface(typeToInject);
174 
175                 if (resolvedType != null && !resolvedType.isInterface()) {
176                     // noinspection AssignmentToMethodParameter
177                     testedClass = new TestedClass(resolvedType, resolvedType);
178                     typeToInject = resolvedType;
179                 }
180             }
181         }
182 
183         if (dependency == null) {
184             dependency = createAndRegisterNewInstance(typeToInject, testedClass, injector, injectionPoint,
185                     injectionProvider);
186         }
187 
188         return dependency;
189     }
190 
191     @Nullable
192     private Object createInstanceOfSupportedInterfaceIfApplicable(@Nonnull TestedClass testedClass,
193             @Nonnull Class<?> typeToInject, @Nonnull InjectionPoint injectionPoint,
194             @Nullable InjectionProvider injectionProvider) {
195         Object dependency = null;
196 
197         if (CommonDataSource.class.isAssignableFrom(typeToInject)) {
198             dependency = createAndRegisterDataSource(testedClass, injectionPoint, injectionProvider);
199         } else if (INJECT_CLASS != null && typeToInject == Provider.class) {
200             assert injectionProvider != null;
201             dependency = createProviderInstance(injectionProvider);
202         } else if (CONVERSATION_CLASS != null && typeToInject == Conversation.class) {
203             dependency = createAndRegisterConversationInstance();
204         } else if (servletDependencies != null && ServletDependencies.isApplicable(typeToInject)) {
205             dependency = servletDependencies.createAndRegisterDependency(typeToInject);
206         } else if (jpaDependencies != null && JPADependencies.isApplicable(typeToInject)) {
207             dependency = jpaDependencies.createAndRegisterDependency(typeToInject, injectionPoint, injectionProvider);
208         }
209 
210         return dependency;
211     }

213     @Nullable
214     private Object createAndRegisterDataSource(@Nonnull TestedClass testedClass, @Nonnull InjectionPoint injectionPoint,
215             @Nullable InjectionProvider injectionProvider) {
216         if (injectionProvider == null || !injectionProvider.hasAnnotation(Resource.class)) {
217             return null;
218         }
219 
220         TestDataSource dsCreation = new TestDataSource(injectionPoint);
221         CommonDataSource dataSource = dsCreation.createIfDataSourceDefinitionAvailable(testedClass);
222 
223         if (dataSource != null) {
224             injectionState.saveInstantiatedDependency(injectionPoint, dataSource);
225         }
226 
227         return dataSource;
228     }

230     @Nonnull
231     private Object createProviderInstance(@Nonnull InjectionProvider injectionProvider) {
232         ParameterizedType genericType = (ParameterizedType) injectionProvider.getDeclaredType();
233         final Class<?> providedClass = (Class<?>) genericType.getActualTypeArguments()[0];
234 
235         if (providedClass.isAnnotationPresent(Singleton.class)) {
236             return new Provider<Object>() {
237                 private Object dependency;
238 
239                 @Override
240                 public synchronized Object get() {
241                     if (dependency == null) {
242                         dependency = createNewInstance(providedClass, true);
243                     }
244 
245                     return dependency;
246                 }
247             };
248         }
249 
250         return (Provider<Object>) () -> createNewInstance(providedClass, false);
251     }

253     @Nullable
254     private Object createNewInstance(@Nonnull Class<?> classToInstantiate, boolean required) {
255         if (classToInstantiate.isInterface()) {
256             return null;
257         }
258 
259         if (classToInstantiate.getClassLoader() == null) {
260             return newInstanceUsingDefaultConstructorIfAvailable(classToInstantiate);
261         }
262 
263         return new TestedObjectCreation(injectionState, this, classToInstantiate).create(required, false);
264     }

266     @Nonnull
267     private Object createAndRegisterConversationInstance() {
268         Conversation conversation = new TestConversation();
269 
270         InjectionPoint injectionPoint = new InjectionPoint(Conversation.class);
271         injectionState.saveInstantiatedDependency(injectionPoint, conversation);
272         return conversation;
273     }

275     @Nullable
276     private Object createAndRegisterNewInstance(@Nonnull Class<?> typeToInstantiate, @Nonnull TestedClass testedClass,
277             @Nonnull Injector injector, @Nonnull InjectionPoint injectionPoint,
278             @Nullable InjectionProvider injectionProvider) {
279         Object dependency = createNewInstance(typeToInstantiate,
280                 injectionProvider != null && injectionProvider.isRequired());
281 
282         if (dependency != null) {
283             if (injectionPoint.name == null) {
284                 assert injectionProvider != null;
285                 injectionPoint = new InjectionPoint(injectionPoint.type, injectionProvider.getName());
286             }
287 
288             registerNewInstance(testedClass, injector, injectionPoint, dependency);
289         }
290 
291         return dependency;
292     }
293 
294     private void registerNewInstance(@Nonnull TestedClass testedClass, @Nonnull Injector injector,
295             @Nonnull InjectionPoint injectionPoint, @Nonnull Object dependency) {
296         injectionState.saveInstantiatedDependency(injectionPoint, dependency);
297 
298         Class<?> instantiatedClass = dependency.getClass();
299 
300         if (testedClass.isClassFromSameModuleOrSystemAsTestedClass(instantiatedClass)) {
301             injector.fillOutDependenciesRecursively(dependency, testedClass);
302             injectionState.lifecycleMethods.findLifecycleMethods(instantiatedClass);
303             injectionState.lifecycleMethods.executeInitializationMethodsIfAny(instantiatedClass, dependency);
304         }
305     }

307     @Override
308     public String toString() {
309         String description = "@Tested object \"" + testedClassName + ' ' + testedName + '"';
310 
311         if (parentInjectionProvider != null) {
312             InjectionProvider injectionProvider = parentInjectionProvider.parent;
313 
314             if (injectionProvider != null) {
315                 description = injectionProvider + "\r\n  of " + description;
316             }
317         }
318 
319         return description;
320     }

322     public void clear() {
323         parentInjectionProvider = null;
324     }
