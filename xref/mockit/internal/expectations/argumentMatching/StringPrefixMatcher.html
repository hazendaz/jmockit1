5   package mockit.internal.expectations.argumentMatching;
6   
7   import edu.umd.cs.findbugs.annotations.NonNull;
8   import edu.umd.cs.findbugs.annotations.Nullable;
9   
10  public final class StringPrefixMatcher extends SubstringMatcher {
11      public StringPrefixMatcher(@NonNull CharSequence substring) {
12          super(substring);
13      }
14  
15      @Override
16      public boolean matches(@Nullable Object argValue) {
17          return argValue instanceof CharSequence && argValue.toString().startsWith(substring);
18      }
19  
20      @Override
21      public void writeMismatchPhrase(@NonNull ArgumentMismatch argumentMismatch) {
22          argumentMismatch.append("a string starting with ").appendFormatted(substring);
23      }
24  }
