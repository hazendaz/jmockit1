5   package mockit.internal.expectations.argumentMatching;
6   
7   import edu.umd.cs.findbugs.annotations.NonNull;
8   import edu.umd.cs.findbugs.annotations.Nullable;
9   
10  public final class NullityMatcher implements ArgumentMatcher<NullityMatcher> {
11      public static final ArgumentMatcher<?> INSTANCE = new NullityMatcher();
12  
13      private NullityMatcher() {
14      }
15  
16      @Override
17      public boolean same(@NonNull NullityMatcher other) {
18          return true;
19      }
20  
21      @Override
22      public boolean matches(@Nullable Object argValue) {
23          return argValue == null;
24      }
25  
26      @Override
27      public void writeMismatchPhrase(@NonNull ArgumentMismatch argumentMismatch) {
28          argumentMismatch.append("null");
29      }
30  }
