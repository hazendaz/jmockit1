35  public final class ReturnTypeConversion {
36      private static final Class<?>[] STRING = { String.class };
37  
38      @NonNull
39      private ExpectedInvocation invocation;
40      @Nullable
41      private final InvocationResults invocationResults;
42      @NonNull
43      private final Class<?> returnType;
44      @NonNull
45      private final Object valueToReturn;
46  
47      public ReturnTypeConversion(@NonNull ExpectedInvocation invocation, @NonNull InvocationResults invocationResults,
48              @NonNull Class<?> returnType, @NonNull Object value) {
49          this.invocation = invocation;
50          this.invocationResults = invocationResults;
51          this.returnType = returnType;
52          valueToReturn = value;
53      }
54  
55      ReturnTypeConversion(@NonNull ExpectedInvocation invocation, @NonNull Class<?> returnType, @NonNull Object value) {
56          this.invocation = invocation;
57          invocationResults = null;
58          this.returnType = returnType;
59          valueToReturn = value;
60      }
61  
62      @NonNull
63      Object getConvertedValue() {
64          Class<?> wrapperType = getWrapperType();
65          Class<?> valueType = valueToReturn.getClass();
66  
67          if (valueType == wrapperType) {
68              return valueToReturn;
69          }
70  
71          if (wrapperType != null && AutoBoxing.isWrapperOfPrimitiveType(valueType)) {
72              return getPrimitiveValueConvertingAsNeeded(wrapperType);
73          }
74  
75          throw newIncompatibleTypesException();
76      }
77  
78      @Nullable
79      private Class<?> getWrapperType() {
80          return AutoBoxing.isWrapperOfPrimitiveType(returnType) ? returnType : AutoBoxing.getWrapperType(returnType);
81      }
82  
83      public void addConvertedValue() {
84          Class<?> wrapperType = getWrapperType();
85          Class<?> valueType = valueToReturn.getClass();
86  
87          if (valueType == wrapperType) {
88              addReturnValue(valueToReturn);
89          } else if (wrapperType != null && AutoBoxing.isWrapperOfPrimitiveType(valueType)) {
90              addPrimitiveValueConvertingAsNeeded(wrapperType);
91          } else {
92              boolean valueIsArray = valueType.isArray();
93  
94              if (valueIsArray || valueToReturn instanceof Iterable<?> || valueToReturn instanceof Iterator<?>) {
95                  assert invocationResults != null;
96                  new MultiValuedConversion(invocationResults, returnType, valueToReturn)
97                          .addMultiValuedResultBasedOnTheReturnType(valueIsArray);
98              } else if (wrapperType != null) {
99                  throw newIncompatibleTypesException();
100             } else {
101                 addResultFromSingleValue();
102             }
103         }
104     }
105 
106     private void addReturnValue(@NonNull Object returnValue) {
107         assert invocationResults != null;
108         invocationResults.addReturnValueResult(returnValue);
109     }
110 
111     private void addPrimitiveValueConvertingAsNeeded(@NonNull Class<?> targetType) {
112         Object convertedValue = getPrimitiveValueConvertingAsNeeded(targetType);
113         addReturnValue(convertedValue);
114     }
115 
116     private void addResultFromSingleValue() {
117         if (returnType == Object.class) {
118             addReturnValue(valueToReturn);
119         } else if (returnType == void.class) {
120             throw newIncompatibleTypesException();
121         } else if (addByteArrayIfApplicable()) {
122             // Do nothing
123         } else if (returnType.isArray()) {
124             addArray();
125         } else if (returnType.isAssignableFrom(ListIterator.class)) {
126             addListIterator();
127         } else if (addCollectionWithSingleElement() || JAVA8 && addJava8ObjectIfApplicable()) {
128             // Do nothing
129         } else if (valueToReturn instanceof CharSequence) {
130             addCharSequence((CharSequence) valueToReturn);
131         } else {
132             addPrimitiveValue();
133         }
134     }
135 
136     @NonNull
137     private IllegalArgumentException newIncompatibleTypesException() {
138         String valueTypeName = JAVA_LANG.matcher(valueToReturn.getClass().getName()).replaceAll("");
139         String returnTypeName = JAVA_LANG.matcher(returnType.getName()).replaceAll("");
140 
141         MethodFormatter methodDesc = new MethodFormatter(invocation.getClassDesc(),
142                 invocation.getMethodNameAndDescription());
143         String msg = "Value of type " + valueTypeName + " incompatible with return type " + returnTypeName + " of "
144                 + methodDesc;
145 
146         return new IllegalArgumentException(msg);
147     }
148 
149     private void addArray() {
150         Object array = Array.newInstance(returnType.getComponentType(), 1);
151         Array.set(array, 0, valueToReturn);
152         addReturnValue(array);
153     }
154 
155     private void addListIterator() {
156         List<Object> l = new ArrayList<>(1);
157         l.add(valueToReturn);
158         ListIterator<Object> iterator = l.listIterator();
159         addReturnValue(iterator);
160     }
161 
162     private void addCharSequence(@NonNull CharSequence textualValue) {
163         @NonNull
164         Object convertedValue = textualValue;
165 
166         if (returnType.isAssignableFrom(ByteArrayInputStream.class)) {
167             // noinspection resource
168             convertedValue = new ByteArrayInputStream(textualValue.toString().getBytes());
169         } else if (returnType.isAssignableFrom(StringReader.class)) {
170             // noinspection resource
171             convertedValue = new StringReader(textualValue.toString());
172         } else if (!(textualValue instanceof StringBuilder) && returnType.isAssignableFrom(StringBuilder.class)) {
173             convertedValue = new StringBuilder(textualValue);
174         } else if (!(textualValue instanceof CharBuffer) && returnType.isAssignableFrom(CharBuffer.class)) {
175             convertedValue = CharBuffer.wrap(textualValue);
176         } else {
177             Object valueFromText = newInstanceUsingPublicConstructorIfAvailable(returnType, STRING, textualValue);
178 
179             if (valueFromText != null) {
180                 convertedValue = valueFromText;
181             }
182         }
183 
184         addReturnValue(convertedValue);
185     }
186 
187     private boolean addByteArrayIfApplicable() {
188         if (returnType == byte[].class && valueToReturn instanceof CharSequence) {
189             addReturnValue(valueToReturn.toString().getBytes());
190             return true;
191         }
192 
193         return false;
194     }
195 
196     private boolean addCollectionWithSingleElement() {
197         Collection<Object> container = null;
198 
199         if (returnType.isAssignableFrom(ArrayList.class)) {
200             container = new ArrayList<>(1);
201         } else if (returnType.isAssignableFrom(LinkedList.class)) {
202             container = new LinkedList<>();
203         } else if (returnType.isAssignableFrom(HashSet.class)) {
204             container = new HashSet<>(1);
205         } else if (returnType.isAssignableFrom(TreeSet.class)) {
206             container = new TreeSet<>();
207         }
208 
209         if (container != null) {
210             container.add(valueToReturn);
211             addReturnValue(container);
212             return true;
213         }
214 
215         return false;
216     }
217 
218     @SuppressWarnings("Since15")
219     private boolean addJava8ObjectIfApplicable() {
220         if (returnType == Optional.class) {
221             addReturnValue(Optional.of(valueToReturn));
222             return true;
223         }
224         if (returnType.isAssignableFrom(Stream.class)) {
225             addReturnValue(singletonList(valueToReturn).stream());
226             return true;
227         }
228 
229         return false;
230     }
231 
232     private void addPrimitiveValue() {
233         Class<?> primitiveType = AutoBoxing.getPrimitiveType(valueToReturn.getClass());
234 
235         if (primitiveType != null) {
236             Class<?>[] parameterType = { primitiveType };
237             Object convertedValue = newInstanceUsingPublicConstructorIfAvailable(returnType, parameterType,
238                     valueToReturn);
239 
240             if (convertedValue == null) {
241                 convertedValue = invokePublicIfAvailable(returnType, null, "valueOf", parameterType, valueToReturn);
242             }
243 
244             if (convertedValue != null) {
245                 addReturnValue(convertedValue);
246                 return;
247             }
248         }
249 
250         throw newIncompatibleTypesException();
251     }
252 
253     @NonNull
254     private Object getPrimitiveValueConvertingAsNeeded(@NonNull Class<?> targetType) {
255         Object convertedValue = null;
256 
257         if (valueToReturn instanceof Number) {
258             convertedValue = convertFromNumber(targetType, (Number) valueToReturn);
259         } else if (valueToReturn instanceof Character) {
260             convertedValue = convertFromChar(targetType, (Character) valueToReturn);
261         }
262 
263         if (convertedValue == null) {
264             throw newIncompatibleTypesException();
265         }
266 
267         return convertedValue;
268     }

270     @Nullable
271     private static Object convertFromNumber(@NonNull Class<?> targetType, @NonNull Number number) {
272         if (targetType == Integer.class) {
273             return number.intValue();
274         }
275 
276         if (targetType == Short.class) {
277             return number.shortValue();
278         }
279 
280         if (targetType == Long.class) {
281             return number.longValue();
282         }
283 
284         if (targetType == Byte.class) {
285             return number.byteValue();
286         }
287 
288         if (targetType == Double.class) {
289             return number.doubleValue();
290         }
291 
292         if (targetType == Float.class) {
293             return number.floatValue();
294         }
295 
296         if (targetType == Character.class) {
297             return (char) number.intValue();
298         }
299 
300         return null;
301     }
302 
303     @Nullable
304     private static Object convertFromChar(@NonNull Class<?> targetType, char c) {
305         if (targetType == Integer.class) {
306             return (int) c;
307         }
308 
309         if (targetType == Short.class) {
310             return (short) c;
311         }
312 
313         if (targetType == Long.class) {
314             return (long) c;
315         }
316 
317         if (targetType == Byte.class) {
318             // noinspection NumericCastThatLosesPrecision
319             return (byte) c;
320         }
321 
322         if (targetType == Double.class) {
323             return (double) c;
324         }
325 
326         if (targetType == Float.class) {
327             return (float) c;
328         }
329 
330         return null;
331     }
