3   import jakarta.persistence.EntityManager;
4   import jakarta.persistence.EntityManagerFactory;
5   import jakarta.persistence.Persistence;
6   import jakarta.persistence.PersistenceException;
7   import jakarta.persistence.Query;

9   import java.sql.SQLException;
10  import java.util.List;
11  
12  /**
13   * This class is a <em>static facade</em> for persistence operations. All methods are <code>static</code>, so it can be
14   * statically imported in client classes for maximum usability.
15   * <p>
16   * All of the persistence operations made available through this facade access a thread-bound <em>persistence
17   * context</em>. In this particular implementation, the standard <strong>JPA</strong> API is used, where
18   * <code>jakarta.persistence.EntityManager</code> represents a work unit. Typically, each work unit instance exists only
19   * long enough to perform a single database transaction. Transaction demarcation is not a responsibility of this class,
20   * however, which simply keeps the association between the current thread and a dedicated work unit object (an
21   * <code>EntityManager</code> instance). (In a web app, the persistence context can be tied to the HTTP request/response
22   * cycle, which normally runs entirely in a single thread for each request/response pair; a central action-dispatch
23   * servlet can commit or rollback the current transaction, while a custom <code>jakarta.servlet.Filter</code> can close
24   * the thread-bound <code>EntityManager</code>.)
25   * <p>
26   * Compared to direct use of an ORM API such as JPA, or to the use of <em>Data Access Objects</em> (the "DAO" pattern),
27   * this <em>static persistence facade</em> pattern has several advantages. Mainly, client code which needs to perform
28   * high-level persistence operations (such as persisting a new entity instance, deleting an already persisted entity, or
29   * finding and loading persistent entities) tends to be simpler and shorter. In the case of entity-specific DAO classes,
30   * potentially thousands of lines of code are saved, without any real loss in portability (the facade implementation can
31   * adapt to new versions of the ORM API or even a different ORM API; and consider that switching between an ORM API and
32   * plain use of JDBC is not viable anyway, since with JDBC the DAO classes need to expose methods for the execution of
33   * "UPDATE" statements, which are not needed with modern ORM APIs).
34   */
35  public final class Database {
36      private static final ThreadLocal<EntityManager> workUnit = new ThreadLocal<>();
37      private static final EntityManagerFactory entityManagerFactory;
38  
39      static {
40          EntityManagerFactory factory = null;
41  
42          try {
43              factory = Persistence.createEntityManagerFactory("AppPersistenceUnit");
44          } catch (PersistenceException e) {
45              e.printStackTrace();
46          }
47  
48          entityManagerFactory = factory;
49      }
50  
51      private Database() {
52      }
53  
54      public static <E> E find(Class<E> entityClass, Object entityId) {
55          return workUnit().find(entityClass, entityId);
56      }
57  
58      public static <E> List<E> find(String ql, Object... args) {
59          Query query = workUnit().createQuery(ql);
60          int position = 1;
61  
62          for (Object arg : args) {
63              query.setParameter(position, arg);
64              position++;
65          }
66  
67          try {
68              return query.getResultList();
69          } catch (PersistenceException e) {
70              Throwable cause = e.getCause();
71              throw cause instanceof SQLException ? new RuntimeException(cause) : e;
72          }
73      }
74  
75      public static void persist(Object data) {
76          workUnit().persist(data);
77      }
78  
79      public static void remove(Object persistentEntity) {
80          workUnit().remove(persistentEntity);
81      }
82  
83      private static EntityManager workUnit() {
84          EntityManager wu = workUnit.get();
85  
86          if (wu == null) {
87              wu = entityManagerFactory.createEntityManager();
88              workUnit.set(wu);
89          }
90  
91          return wu;
92      }
93  }
