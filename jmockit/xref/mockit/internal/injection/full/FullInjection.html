72      public FullInjection(@NonNull InjectionState injectionState, @NonNull Class<?> testedClass,
73              @NonNull String testedName) {
74          this.injectionState = injectionState;
75          testedClassName = testedClass.getSimpleName();
76          this.testedName = testedName;
77          servletJakartaDependencies = JAKARTA_SERVLET_CLASS == null ? null
78                  : new ServletJakartaDependencies(injectionState);
79          servletJavaxDependencies = JAVAX_SERVLET_CLASS == null ? null : new ServletJavaxDependencies(injectionState);
80          jpaJakartaDependencies = JAKARTA_PERSISTENCE_UNIT_CLASS == null ? null
81                  : new JPAJakartaDependencies(injectionState);
82          jpaJavaxDependencies = JAVAX_PERSISTENCE_UNIT_CLASS == null ? null : new JPAJavaxDependencies(injectionState);
83      }

85      @Nullable
86      public Object createOrReuseInstance(@NonNull TestedClass testedClass, @NonNull Injector injector,
87              @Nullable InjectionProvider injectionProvider, @Nullable String qualifiedName) {
88          setInjectionProvider(injectionProvider);
89  
90          InjectionPoint injectionPoint = getInjectionPoint(testedClass, injectionProvider, qualifiedName);
91          Object dependency = injectionState.getInstantiatedDependency(testedClass, injectionPoint);
92  
93          if (dependency != null) {
94              return dependency;
95          }
96  
97          Class<?> typeToInject = dependencyClass;
98  
99          if (typeToInject == Logger.class) {
100             return createLogger(testedClass);
101         }
102 
103         if (typeToInject == null || !isInstantiableType(typeToInject)) {
104             return null;
105         }
106 
107         return createInstance(testedClass, injector, injectionProvider, injectionPoint);
108     }

110     public void setInjectionProvider(@Nullable InjectionProvider injectionProvider) {
111         if (injectionProvider != null) {
112             injectionProvider.parent = parentInjectionProvider;
113         }
114 
115         parentInjectionProvider = injectionProvider;
116     }

118     @NonNull
119     private InjectionPoint getInjectionPoint(@NonNull TestedClass testedClass,
120             @Nullable InjectionProvider injectionProvider, @Nullable String qualifiedName) {
121         if (injectionProvider == null) {
122             dependencyClass = testedClass.targetClass;
123             return new InjectionPoint(dependencyClass, qualifiedName, true);
124         }
125 
126         Type dependencyType = injectionProvider.getDeclaredType();
127 
128         if (dependencyType instanceof TypeVariable<?>) {
129             dependencyType = testedClass.reflection.resolveTypeVariable((TypeVariable<?>) dependencyType);
130             dependencyClass = getClassType(dependencyType);
131         } else {
132             dependencyClass = injectionProvider.getClassOfDeclaredType();
133         }
134 
135         if (qualifiedName != null && !qualifiedName.isEmpty()) {
136             return new InjectionPoint(dependencyClass, qualifiedName, true);
137         }
138 
139         if (jpaJakartaDependencies != null && JPAJakartaDependencies.isApplicable(dependencyClass)) {
140             for (Annotation annotation : injectionProvider.getAnnotations()) {
141                 InjectionPoint injectionPoint = jpaJakartaDependencies.getInjectionPointIfAvailable(annotation);
142 
143                 if (injectionPoint != null) {
144                     return injectionPoint;
145                 }
146             }
147         }
148 
149         if (jpaJavaxDependencies != null && JPAJavaxDependencies.isApplicable(dependencyClass)) {
150             for (Annotation annotation : injectionProvider.getAnnotations()) {
151                 InjectionPoint injectionPoint = jpaJavaxDependencies.getInjectionPointIfAvailable(annotation);
152 
153                 if (injectionPoint != null) {
154                     return injectionPoint;
155                 }
156             }
157         }
158 
159         return new InjectionPoint(dependencyType, injectionProvider.getName(), false);
160     }

162     @NonNull
163     private static Object createLogger(@NonNull TestedClass testedClass) {
164         TestedClass testedClassWithLogger = testedClass.parent;
165         assert testedClassWithLogger != null;
166         return Logger.getLogger(testedClassWithLogger.nameOfTestedClass);
167     }

169     public static boolean isInstantiableType(@NonNull Class<?> type) {
170         if (type.isPrimitive() || type.isArray() || type.isAnnotation()) {
171             return false;
172         }
173 
174         if (type.isInterface()) {
175             return true;
176         }
177 
178         int typeModifiers = type.getModifiers();
179 
180         if ((typeModifiers & INVALID_TYPES) != 0 || !isStatic(typeModifiers) && type.isMemberClass()) {
181             return false;
182         }
183 
184         return type.getClassLoader() != null;
185     }

187     @Nullable
188     private Object createInstance(@NonNull TestedClass testedClass, @NonNull Injector injector,
189             @Nullable InjectionProvider injectionProvider, @NonNull InjectionPoint injectionPoint) {
190         @SuppressWarnings("ConstantConditions")
191         @NonNull
192         Class<?> typeToInject = dependencyClass;
193         Object dependency = null;
194 
195         if (typeToInject.isInterface()) {
196             dependency = createInstanceOfSupportedInterfaceIfApplicable(testedClass, typeToInject, injectionPoint,
197                     injectionProvider);
198 
199             if (dependency == null && typeToInject.getClassLoader() != null) {
200                 Class<?> resolvedType = injectionState.resolveInterface(typeToInject);
201 
202                 if (resolvedType != null && !resolvedType.isInterface()) {
203                     // noinspection AssignmentToMethodParameter
204                     testedClass = new TestedClass(resolvedType, resolvedType);
205                     typeToInject = resolvedType;
206                 }
207             }
208         }
209 
210         if (dependency == null) {
211             dependency = createAndRegisterNewInstance(typeToInject, testedClass, injector, injectionPoint,
212                     injectionProvider);
213         }
214 
215         return dependency;
216     }

218     @Nullable
219     private Object createInstanceOfSupportedInterfaceIfApplicable(@NonNull TestedClass testedClass,
220             @NonNull Class<?> typeToInject, @NonNull InjectionPoint injectionPoint,
221             @Nullable InjectionProvider injectionProvider) {
222         Object dependency = null;
223 
224         if (CommonDataSource.class.isAssignableFrom(typeToInject)) {
225             dependency = createAndRegisterDataSource(testedClass, injectionPoint, injectionProvider);
226         } else if (JAKARTA_INJECT_CLASS != null && typeToInject == jakarta.inject.Provider.class) {
227             assert injectionProvider != null;
228             dependency = createProviderJakartaInstance(injectionProvider);
229         } else if (JAVAX_INJECT_CLASS != null && typeToInject == javax.inject.Provider.class) {
230             assert injectionProvider != null;
231             dependency = createProviderJavaxInstance(injectionProvider);
232         } else if (JAKARTA_CONVERSATION_CLASS != null
233                 && typeToInject == jakarta.enterprise.context.Conversation.class) {
234             dependency = createAndRegisterConversationJakartaInstance();
235         } else if (JAVAX_CONVERSATION_CLASS != null && typeToInject == javax.enterprise.context.Conversation.class) {
236             dependency = createAndRegisterConversationJavaxInstance();
237         } else if (servletJakartaDependencies != null && ServletJakartaDependencies.isApplicable(typeToInject)) {
238             dependency = servletJakartaDependencies.createAndRegisterDependency(typeToInject);
239         } else if (servletJavaxDependencies != null && ServletJavaxDependencies.isApplicable(typeToInject)) {
240             dependency = servletJavaxDependencies.createAndRegisterDependency(typeToInject);
241         } else if (jpaJakartaDependencies != null && JPAJakartaDependencies.isApplicable(typeToInject)) {
242             dependency = jpaJakartaDependencies.createAndRegisterDependency(typeToInject, injectionPoint,
243                     injectionProvider);
244         } else if (jpaJavaxDependencies != null && JPAJavaxDependencies.isApplicable(typeToInject)) {
245             dependency = jpaJavaxDependencies.createAndRegisterDependency(typeToInject, injectionPoint,
246                     injectionProvider);
247         }
248 
249         return dependency;
250     }

252     @Nullable
253     private Object createAndRegisterDataSource(@NonNull TestedClass testedClass, @NonNull InjectionPoint injectionPoint,
254             @Nullable InjectionProvider injectionProvider) {
255         if (injectionProvider == null) {
256             return null;
257         }
258 
259         // Check annotation is present (both jars)
260         if ((JAKARTA_RESOURCE_CLASS != null && injectionProvider.hasAnnotation(jakarta.annotation.Resource.class))
261                 || (JAVAX_RESOURCE_CLASS != null && injectionProvider.hasAnnotation(javax.annotation.Resource.class))) {
262             TestDataSource dsCreation = new TestDataSource(injectionPoint);
263             CommonDataSource dataSource = dsCreation.createIfDataSourceDefinitionAvailable(testedClass);
264 
265             if (dataSource != null) {
266                 injectionState.saveInstantiatedDependency(injectionPoint, dataSource);
267             }
268 
269             return dataSource;
270         }
271 
272         return null;
273     }

275     @NonNull
276     private Object createProviderJakartaInstance(@NonNull InjectionProvider injectionProvider) {
277         ParameterizedType genericType = (ParameterizedType) injectionProvider.getDeclaredType();
278         final Class<?> providedClass = (Class<?>) genericType.getActualTypeArguments()[0];
279 
280         if (providedClass.isAnnotationPresent(jakarta.inject.Singleton.class)) {
281             return new jakarta.inject.Provider<Object>() {
282                 private Object dependency;
283 
284                 @Override
285                 public synchronized Object get() {
286                     if (dependency == null) {
287                         dependency = createNewInstance(providedClass, true);
288                     }
289 
290                     return dependency;
291                 }
292             };
293         }
294 
295         return (jakarta.inject.Provider<Object>) () -> createNewInstance(providedClass, false);
296     }

298     @NonNull
299     private Object createProviderJavaxInstance(@NonNull InjectionProvider injectionProvider) {
300         ParameterizedType genericType = (ParameterizedType) injectionProvider.getDeclaredType();
301         final Class<?> providedClass = (Class<?>) genericType.getActualTypeArguments()[0];
302 
303         if (providedClass.isAnnotationPresent(javax.inject.Singleton.class)) {
304             return new javax.inject.Provider<Object>() {
305                 private Object dependency;
306 
307                 @Override
308                 public synchronized Object get() {
309                     if (dependency == null) {
310                         dependency = createNewInstance(providedClass, true);
311                     }
312 
313                     return dependency;
314                 }
315             };
316         }
317 
318         return (javax.inject.Provider<Object>) () -> createNewInstance(providedClass, false);
319     }

321     @Nullable
322     private Object createNewInstance(@NonNull Class<?> classToInstantiate, boolean required) {
323         if (classToInstantiate.isInterface()) {
324             return null;
325         }
326 
327         if (classToInstantiate.getClassLoader() == null) {
328             return newInstanceUsingDefaultConstructorIfAvailable(classToInstantiate);
329         }
330 
331         return new TestedObjectCreation(injectionState, this, classToInstantiate).create(required, false);
332     }

352     @Nullable
353     private Object createAndRegisterNewInstance(@NonNull Class<?> typeToInstantiate, @NonNull TestedClass testedClass,
354             @NonNull Injector injector, @NonNull InjectionPoint injectionPoint,
355             @Nullable InjectionProvider injectionProvider) {
356         Object dependency = createNewInstance(typeToInstantiate,
357                 injectionProvider != null && injectionProvider.isRequired());
358 
359         if (dependency != null) {
360             if (injectionPoint.name == null) {
361                 assert injectionProvider != null;
362                 injectionPoint = new InjectionPoint(injectionPoint.type, injectionProvider.getName());
363             }
364 
365             registerNewInstance(testedClass, injector, injectionPoint, dependency);
366         }
367 
368         return dependency;
369     }
370 
371     private void registerNewInstance(@NonNull TestedClass testedClass, @NonNull Injector injector,
372             @NonNull InjectionPoint injectionPoint, @NonNull Object dependency) {
373         injectionState.saveInstantiatedDependency(injectionPoint, dependency);
374 
375         Class<?> instantiatedClass = dependency.getClass();
376 
377         if (testedClass.isClassFromSameModuleOrSystemAsTestedClass(instantiatedClass)) {
378             injector.fillOutDependenciesRecursively(dependency, testedClass);
379             injectionState.lifecycleMethods.findLifecycleMethods(instantiatedClass);
380             injectionState.lifecycleMethods.executeInitializationMethodsIfAny(instantiatedClass, dependency);
381         }
382     }

384     @Override
385     public String toString() {
386         String description = "@Tested object \"" + testedClassName + ' ' + testedName + '"';
387 
388         if (parentInjectionProvider != null) {
389             InjectionProvider injectionProvider = parentInjectionProvider.parent;
390 
391             if (injectionProvider != null) {
392                 description = injectionProvider + "\r\n  of " + description;
393             }
394         }
395 
396         return description;
397     }
