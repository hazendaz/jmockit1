5   package mockit.internal.state;
6   
7   import edu.umd.cs.findbugs.annotations.NonNull;
8   import edu.umd.cs.findbugs.annotations.Nullable;
9   
10  import java.lang.instrument.ClassFileTransformer;
11  import java.security.ProtectionDomain;
12  import java.util.HashMap;
13  import java.util.Map;
14  import java.util.WeakHashMap;
15  
16  import mockit.internal.startup.Startup;
17  
18  /**
19   * Holds a map of internal class names to the corresponding class files (bytecode arrays), for the classes that have
20   * already been loaded during the test run. These classfiles are not necessarily the same as those stored in the
21   * corresponding ".class" files available from the runtime classpath. If any third-party {@link ClassFileTransformer}s
22   * are active, those original classfiles may have been modified before being loaded by the JVM. JMockit installs a
23   * <code>ClassFileTransformer</code> of its own which saves all potentially modified classfiles here.
24   * <p>
25   * This bytecode cache allows classes to be mocked and un-mocked correctly, even in the presence of other bytecode
26   * modification agents such as the AspectJ load-time weaver.
27   */
28  public final class CachedClassfiles implements ClassFileTransformer {
29      @NonNull
30      public static final CachedClassfiles INSTANCE = new CachedClassfiles();
31  
32      @NonNull
33      private final Map<ClassLoader, Map<String, byte[]>> classLoadersAndClassfiles;
34      @Nullable
35      private Class<?> classBeingCached;
36  
37      private CachedClassfiles() {
38          classLoadersAndClassfiles = new WeakHashMap<>(2);
39      }
40  
41      @Nullable
42      @Override
43      public byte[] transform(@Nullable ClassLoader loader, String classDesc,
44              @Nullable Class<?> classBeingRedefinedOrRetransformed, @Nullable ProtectionDomain protectionDomain,
45              @NonNull byte[] classfileBuffer) {
46          // can be null for Java 8 lambdas
47          if (classDesc != null && classBeingRedefinedOrRetransformed != null
48                  && classBeingRedefinedOrRetransformed == classBeingCached) {
49              addClassfile(loader, classDesc, classfileBuffer);
50              classBeingCached = null;
51          }
52  
53          return null;
54      }
55  
56      private void addClassfile(@Nullable ClassLoader loader, @NonNull String classDesc, @NonNull byte[] classfile) {
57          Map<String, byte[]> classfiles = getClassfiles(loader);
58          classfiles.put(classDesc, classfile);
59      }
60  
61      @NonNull
62      private Map<String, byte[]> getClassfiles(@Nullable ClassLoader loader) {
63          return classLoadersAndClassfiles.computeIfAbsent(loader, k -> new HashMap<>(100));
64      }
65  
66      @Nullable
67      private byte[] findClassfile(@NonNull Class<?> aClass) {
68          String className = aClass.getName();
69  
70          // Discards an invalid numerical suffix from a synthetic Java 8 class, if detected.
71          int p = className.indexOf('/');
72          if (p > 0) {
73              className = className.substring(0, p);
74          }
75  
76          Map<String, byte[]> classfiles = getClassfiles(aClass.getClassLoader());
77          return classfiles.get(className.replace('.', '/'));
78      }
79  
80      @Nullable
81      public static synchronized byte[] getClassfile(@NonNull String classDesc) {
82          return INSTANCE.findClassfile(classDesc);
83      }
84  
85      @Nullable
86      private byte[] findClassfile(@NonNull String classDesc) {
87          byte[] classfile = null;
88  
89          for (Map<String, byte[]> classfiles : classLoadersAndClassfiles.values()) {
90              classfile = classfiles.get(classDesc);
91  
92              if (classfile != null) {
93                  return classfile;
94              }
95          }
96  
97          Class<?> desiredClass = Startup.getClassIfLoaded(classDesc);
98  
99          if (desiredClass != null) {
100             classBeingCached = desiredClass;
101             Startup.retransformClass(desiredClass);
102             ClassLoader classLoader = desiredClass.getClassLoader();
103             classfile = INSTANCE.findClassfile(classLoader, classDesc);
104         }
105 
106         return classfile;
107     }
108 
109     @Nullable
110     private synchronized byte[] findClassfile(@Nullable ClassLoader loader, @NonNull String classDesc) {
111         Map<String, byte[]> classfiles = getClassfiles(loader);
112         return classfiles.get(classDesc);
113     }
114 
115     @Nullable
116     public static synchronized byte[] getClassfile(@NonNull Class<?> aClass) {
117         byte[] cached = INSTANCE.findClassfile(aClass);
118         if (cached != null) {
119             return cached;
120         }
121 
122         INSTANCE.classBeingCached = aClass;
123         Startup.retransformClass(aClass);
124         return INSTANCE.findClassfile(aClass);
125     }
126 
127     @Nullable
128     public static byte[] getClassfile(@Nullable ClassLoader loader, @NonNull String internalClassName) {
129         return INSTANCE.findClassfile(loader, internalClassName);
130     }
131 
132     public static void addClassfile(@NonNull Class<?> aClass, @NonNull byte[] classfile) {
133         INSTANCE.addClassfile(aClass.getClassLoader(), aClass.getName().replace('.', '/'), classfile);
134     }
135 }
