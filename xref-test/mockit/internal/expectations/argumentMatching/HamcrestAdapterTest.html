6   package mockit.internal.expectations.argumentMatching;
7   
8   import static org.junit.jupiter.api.Assertions.assertEquals;
9   import static org.junit.jupiter.api.Assertions.assertFalse;
10  import static org.junit.jupiter.api.Assertions.assertNotNull;
11  import static org.junit.jupiter.api.Assertions.assertTrue;
12  
13  import org.hamcrest.core.Is;
14  import org.hamcrest.core.IsEqual;
15  import org.hamcrest.core.IsNot;
16  import org.hamcrest.core.IsSame;
17  import org.junit.jupiter.api.Test;
18  
19  final class HamcrestAdapterTest {
20  
21      @Test
22      void matchesReturnsTrueWhenMatcherMatches() {
23          HamcrestAdapter adapter = new HamcrestAdapter(IsEqual.equalTo("hello"));
24          assertTrue(adapter.matches("hello"));
25      }
26  
27      @Test
28      void matchesReturnsFalseWhenMatcherDoesNotMatch() {
29          HamcrestAdapter adapter = new HamcrestAdapter(IsEqual.equalTo("hello"));
30          assertFalse(adapter.matches("world"));
31      }
32  
33      @Test
34      void sameReturnsTrueForSameMatcher() {
35          org.hamcrest.Matcher<String> matcher = IsEqual.equalTo("hello");
36          HamcrestAdapter adapter1 = new HamcrestAdapter(matcher);
37          HamcrestAdapter adapter2 = new HamcrestAdapter(matcher);
38          assertTrue(adapter1.same(adapter2));
39      }
40  
41      @Test
42      void sameReturnsFalseForDifferentMatcher() {
43          HamcrestAdapter adapter1 = new HamcrestAdapter(IsEqual.equalTo("hello"));
44          HamcrestAdapter adapter2 = new HamcrestAdapter(IsEqual.equalTo("hello"));
45          assertFalse(adapter1.same(adapter2));
46      }
47  
48      @Test
49      void writeMismatchPhraseAppendsMatcherDescription() {
50          HamcrestAdapter adapter = new HamcrestAdapter(IsEqual.equalTo("hello"));
51          ArgumentMismatch mismatch = new ArgumentMismatch();
52          adapter.writeMismatchPhrase(mismatch);
53          String result = mismatch.toString();
54          assertNotNull(result);
55          assertTrue(result.contains("hello"));
56      }
57  
58      @Test
59      void writeMismatchPhraseWithNotMatcher() {
60          HamcrestAdapter adapter = new HamcrestAdapter(IsNot.not("hello"));
61          ArgumentMismatch mismatch = new ArgumentMismatch();
62          adapter.writeMismatchPhrase(mismatch);
63          String result = mismatch.toString();
64          assertNotNull(result);
65          assertFalse(result.isEmpty());
66      }
67  
68      @Test
69      void getInnerValueFromIsEqualMatcher() {
70          HamcrestAdapter adapter = new HamcrestAdapter(IsEqual.equalTo("hello"));
71          Object innerValue = adapter.getInnerValue();
72          assertEquals("hello", innerValue);
73      }
74  
75      @Test
76      void getInnerValueFromIsSameMatcher() {
77          String target = "hello";
78          HamcrestAdapter adapter = new HamcrestAdapter(IsSame.sameInstance(target));
79          Object innerValue = adapter.getInnerValue();
80          assertEquals(target, innerValue);
81      }
82  
83      @Test
84      void getInnerValueFromIsWrapperExtractsInnerValue() {
85          HamcrestAdapter adapter = new HamcrestAdapter(Is.is("hello"));
86          Object innerValue = adapter.getInnerValue();
87          assertEquals("hello", innerValue);
88      }
89  
90      @Test
91      void getInnerValueFromIsNotWrapperUnwrapsInnerMatcher() {
92          HamcrestAdapter adapter = new HamcrestAdapter(IsNot.not("hello"));
93          Object innerValue = adapter.getInnerValue();
94          assertEquals("hello", innerValue);
95      }
96  
97      @Test
98      void getInnerValueFromNestedIsWrappersExtractsInnerValue() {
99          HamcrestAdapter adapter = new HamcrestAdapter(Is.is(Is.is("hello")));
100         Object innerValue = adapter.getInnerValue();
101         assertEquals("hello", innerValue);
102     }
103 
104     @Test
105     void matchesNullArgument() {
106         HamcrestAdapter adapter = new HamcrestAdapter(IsEqual.equalTo(null));
107         assertTrue(adapter.matches(null));
108     }
109 }
