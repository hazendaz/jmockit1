3   import edu.umd.cs.findbugs.annotations.NonNull;
4   import edu.umd.cs.findbugs.annotations.Nullable;
5   
6   import java.util.Date;
7   
8   import javax.inject.Inject;
9   
10  import petclinic.owners.Owner;
11  import petclinic.owners.OwnerData;
12  import petclinic.util.TestDatabase;
13  
14  /**
15   * Utility class for creation of {@link Pet} data in the test database, to be used in integration tests.
16   */
17  public final class PetData extends TestDatabase {
18      @Inject
19      private OwnerData ownerData;
20  
21      @NonNull
22      public Pet findOrCreate(@NonNull String name, @Nullable Date birthDate, @NonNull String petType) {
23          Pet pet = findOne("select p from Pet p where p.name = ?1", name);
24  
25          if (pet == null) {
26              pet = create(name, birthDate, petType);
27          }
28  
29          return pet;
30      }
31  
32      @NonNull
33      public Pet create(@NonNull String name, @Nullable Date birthDate, @NonNull String petType) {
34          Owner owner = ownerData.create("Pet Owner");
35          return create(owner, name, birthDate, petType);
36      }
37  
38      @NonNull
39      public Pet create(@NonNull Owner owner, @NonNull String name, @Nullable Date birthDate, @NonNull String petType) {
40          PetType type = findOrCreatePetType(petType);
41  
42          Pet pet = new Pet();
43          owner.addPet(pet);
44          pet.setName(name);
45          pet.setBirthDate(birthDate);
46          pet.setType(type);
47  
48          db.save(pet);
49          return pet;
50      }
51  
52      @NonNull
53      PetType findOrCreatePetType(@NonNull String petType) {
54          PetType type = findOne("select t from PetType t where t.name = ?1", petType);
55  
56          if (type == null) {
57              type = createType(petType);
58          }
59  
60          return type;
61      }
62  
63      @NonNull
64      PetType createType(@NonNull String name) {
65          PetType type = new PetType();
66          type.setName(name);
67          db.save(type);
68          return type;
69      }
70  }
