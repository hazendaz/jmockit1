6   package petclinic.owners;
7   
8   import edu.umd.cs.findbugs.annotations.NonNull;
9   
10  import petclinic.util.TestDatabase;
11  
12  /**
13   * Utility class for creation of {@link Owner} data in the test database, to be used in integration tests.
14   */
15  public final class OwnerData extends TestDatabase {
16      @NonNull
17      public Owner create(@NonNull String fullName) {
18          String[] names = fullName.split(" ");
19  
20          Owner owner = new Owner();
21          owner.setFirstName(names[0]);
22          owner.setLastName(names[names.length - 1]);
23          owner.setTelephone("01234-5678");
24  
25          db.save(owner);
26          return owner;
27      }
28  }
